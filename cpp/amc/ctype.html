 35void amc::tfunc_Ctype_Uninit() {
 36    // de-initialization order:
 37    // - cascdel fields  in reverse order
 38    // - any unred_body text created by generators.
 39    // - unref fields in reverse order (remove x-references)
 40    // - user-specified cleanup in reverse order
 41    // - un-init fields in reverse order
 42    algo_lib::Replscope R;
 43    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
 44
 45    tempstr uninit_text;
 46    tempstr unref_text;
 47    tempstr refname = amc::Refname(ctype);
 48    bool inl = true;
 49    bool glob = GlobalQ(ctype);
 50    Set(R, "$Ctype"  , ctype.ctype);
 51
 52    // Cascade-delete fields
 53    amc::c_tempfield_RemoveAll();
 54    ind_beg(ctype_c_field_curs,field,ctype) {
 55        c_tempfield_Insert(field);
 56    }ind_end;
 57    rrep_(i, amc::c_tempfield_N()) {
 58        amc::FField &field = *amc::c_tempfield_Find(i);
 59        if (field.c_cascdel) {
 60            Set(R, "$field", field.field);
 61            Set(R, "$name", name_Get(field));
 62            Set(R, "$pararg", glob ? strptr("") : refname);
 63            Ins(&R, unref_text, "$name_Cascdel($pararg); // dmmeta.cascdel:$field");
 64        }
 65    }
 66
 67    // Grab parent's Unref text
 68    if (amc::FFunc* unref = amc::ind_func_Find(Subst(R,"$Ctype..Unref"))) {
 69        unref_text << unref->body;
 70    }
 71
 72    // Unref fields in reverse order
 73    rrep_(i, amc::c_tempfield_N()) {
 74        amc::FField &field = *c_tempfield_Find(i);
 75        if (field.c_fcleanup) {
 76            Set(R, "$field", field.field);
 77            Set(R, "$name", name_Get(field));
 78            Set(R, "$pararg", glob ? strptr("") : refname);
 79            Ins(&R, unref_text, "$name_Cleanup($pararg); // dmmeta.fcleanup:$field");
 80        }
 81        if (amc::FFunc *func = amc::ind_func_Find(dmmeta::Func_Concat_field_name(field.field,"Unref"))) {
 82            if (ch_N(func->body)) {
 83                Set(R, "$field", field.field);
 84                Ins(&R, unref_text, "");
 85                Ins(&R, unref_text, tempstr()<<"// $field.Unref ("<<field.reftype<<")  //"<<field.comment);
 86                unref_text << func->body;
 87                inl = inl && func->inl;
 88            }
 89        }
 90    }
 91
 92    // Uninit fields in reverse order (release memory) -- ALL after user-specified cleanup
 93    rrep_(i, amc::c_tempfield_N()) {
 94        amc::FField &field = *c_tempfield_Find(i);
 95        if (amc::FFunc *func = amc::ind_func_Find(dmmeta::Func_Concat_field_name(field.field,"Uninit"))) {
 96            if (ch_N(func->body)) {
 97                Set(R, "$field", field.field);
 98                Ins(&R, uninit_text, "");
 99                Ins(&R, uninit_text, tempstr()<<"// $field.Uninit ("<<field.reftype<<")  //"<<field.comment);
100                uninit_text << func->body;
101                inl = inl && func->inl;
102            }
103        }
104    }
105
106    if (ch_N(unref_text)) {
107        inl = false;
108    }
109
110    if (ch_N(uninit_text) || ch_N(unref_text) || glob) {
111        Set(R, "$thisparname", glob ? strptr("_db") : refname);
112        Set(R, "$Parent" , glob ? strptr("")    : strptr(tempstr() << ctype.cpp_type << "& " << refname));
113        Set(R, "$Cpptype", ctype.cpp_type);
114        Set(R, "$Name"   , name_Get(ctype));
115        amc::FFunc& uninit = amc::CreateCurFunc();
116        Ins(&R, uninit.ret  , "void", false);
117        Ins(&R, uninit.proto, "$Name_Uninit($Parent)", false);
118        Ins(&R, uninit.body, "$Cpptype &row = $thisparname; (void)row;");
119        // init function MUST be inline at the moment.
120        uninit.inl = inl && (!ctype.c_init || ctype.c_init->inl) && c_tempfield_N() < 10;
121        uninit.body << unref_text;
122        uninit.body << uninit_text;
123    };
124}

128static void CheckSeen(amc::FGenXrefSeen &seen, amc::FXref &xref) {
129    // computed pointer is a function of the path.
130    // check that cached path is the same.
131    // the key we use is just a puny variable name (name of xreffld) but the actual path must be considered,
132    // it consists of xreffld & via
133    amc::FField     *xreffld = GetKeyfld(xref);
134    amc::FField *viafld = GetViafld(xref);
135    if (viafld && seen.via != viafld->field) {
136        prerr("amc.badvia"
137              <<Keyval("field",xref.field)
138              <<Keyval("xreffld",strptr(xreffld->field))
139              <<Keyval("via1",viafld->field)
140              <<Keyval("via2",seen.via)
141              <<Keyval("comment","Lookup of same target via 2 different paths is not allowed"));
142    }
143    if (seen.xreffld != xreffld->field) {
144        prerr("amc.badxreffld"
145              <<Keyval("field",xref.field)
146              <<Keyval("xreffld",xreffld->field)
147              <<Keyval("xreffld2",seen.xreffld)
148              <<Keyval("comment","Lookup of same target via 2 different paths is not allowed"));
149    }
150}

155static void ComputeAccess_CheckNull(algo_lib::Replscope &R, amc::FCtype &ctype, amc::FXref &xref, amc::FFunc &func) {
156    amc::FField     *xreffld = GetKeyfld(xref);
157    amc::FField *viafld = GetViafld(xref);
158    if (xreffld->reftype == dmmeta_Reftype_reftype_Upptr) {
159        Ins(&R, func.body, "if (UNLIKELY(!p_$parent)) {");
160        Ins(&R, func.body, "    algo_lib::ResetErrtext() << \"$ns.null_ref  xref:$xrefkey\";");
161        Ins(&R, func.body, "    return false;");
162        Ins(&R, func.body, "}");
163    } else if (viafld) {
164        Set(R, "$via",viafld->field);
165        Set(R, "$getrow", FieldvalExpr(&ctype, *xreffld, "row"));
166        Ins(&R, func.body, "if (UNLIKELY(!p_$parent)) {");
167        Ins(&R, func.body, "    algo_lib::ResetErrtext() << \"$ns.bad_xref  index:$via\" << Keyval(\"key\", $getrow);");
168        Ins(&R, func.body, "    return false;");
169        Ins(&R, func.body, "}");
170    } else {
171        prerr("amc.xref"
172              <<Keyval("field",xref.field)
173              <<Keyval("comment","xrefvia required"));
174        algo_lib::_db.exit_code++;
175    }

180// Introduce local variable that points to the parent side of XREF
181// by evaluating the path provided by xref + xreffld + xrefvia records.
182static bool ComputeAccess(algo_lib::Replscope &R, amc::FCtype &ctype, amc::FXref &xref, amc::FFunc &func, amc::FGenXref &frame, bool check_null) {
183    bool retval=false;
184    amc::FField *keyfld = GetKeyfld(xref);
185    amc::FField *viafld = GetViafld(xref);
186    if (keyfld) {
187        dmmeta::FieldPkey xreffldname(name_Get(*keyfld));
188        dmmeta::FieldPkey via(viafld ? strptr(viafld->field) : strptr());
189        Set(R, "$parent", xreffldname);
190        Set(R, "$access", FieldvalExpr(&ctype, *keyfld, "row"));// execute early to allow more error checking
191        amc::FGenXrefSeen *seen = ind_seen_Find(frame,xreffldname);
192        if (seen) {
193            CheckSeen(*seen,xref);
194        }
195        if (!seen && !(xref.p_field->p_ctype == ctype.p_ns->c_globfld->p_ctype)) {// ignore global xrefs
196            seen = &seen_Alloc(frame);
197            seen->value = xreffldname;
198            seen->via = via;
199            seen->xreffld = keyfld->field;
200            ind_seen_InsertMaybe(frame,*seen_Last(frame));
201            amc::FCtype          *parentrow = xref.p_field->p_ctype == &ctype ? xref.p_field->p_arg : xref.p_field->p_ctype;
202            if (!(keyfld->reftype == dmmeta_Reftype_reftype_Ptr || keyfld->reftype == dmmeta_Reftype_reftype_Upptr)) {
203                vrfy(viafld,"Missing xrefvia");
204                Set(R, "$xrefvia", tempstr() << ns_Get(*viafld->p_ctype) << "::" << name_Get(*viafld));
205                Set(R, "$access", "$xrefvia_Find($access)");
206            }
207            Set(R, "$ParType", parentrow->cpp_type);
208            Ins(&R, func.body, "$ParType* p_$parent = $access;");
209            if (check_null) {
210                ComputeAccess_CheckNull(R,ctype,xref,func);
211            }
212            retval=true;
213        }
214    } else {
215        Set(R,"$parent","");
216    }
217    return retval;
218}

229static void Ctype_Xref(amc::FCtype &ctype) {
230    algo_lib::Replscope R;
231    amc::FNs &ns = *ctype.p_ns;
232    amc::FField &inst = *FirstInst(ctype);
233    amc::FGenXref frame;
234    Set(R, "$Cpptype"  , ctype.cpp_type);
235    Set(R, "$inst"  , inst.field);
236    Set(R, "$name" , name_Get(inst));
237    Set(R, "$Name" , name_Get(ctype));
238    Set(R, "$Ctype" , ctype.ctype);
239    // xref function for parent field??
240    amc::FFunc& xrefmaybe = amc::ind_func_GetOrCreate(Subst(R,"$inst.XrefMaybe"));
241    Ins(&R, xrefmaybe.comment, "Insert row into all appropriate indices. If error occurs, store error");
242    Ins(&R, xrefmaybe.comment, "in algo_lib::_db.errtext and return false. Caller must Delete or Unref such row.");
243    Ins(&R, xrefmaybe.ret  , "bool", false);
244    Ins(&R, xrefmaybe.body , "bool retval = true;");
245    if (!GlobalQ(ctype)) {
246        Ins(&R, xrefmaybe.body , "(void)row;");
247    }
248    Ins(&R, xrefmaybe.proto, (GlobalQ(ctype) ? "$name_XrefMaybe()" : "$name_XrefMaybe($Cpptype &row)"), false);
249    bool sort_xrefs = ns.c_nsx && ns.c_nsx->sortxref;
250
251    frep_(iter,2) {
252        ind_beg(amc::ctype_zs_xref_curs, xref, ctype) if (xref.inscond.value != "false" && ProcessXrefQ(xref,iter,sort_xrefs)) {
253            amc::FField *keyfld = GetKeyfld(xref);
254            bool upptr = xref.p_field->reftype == dmmeta_Reftype_reftype_Upptr;
255            Set(R, "$xrefname", name_Get(xref));
256            Set(R, "$xrefkey", xref.field);
257            Set(R, "$ns"  , ns.ns);
258            ComputeAccess(R,ctype,xref,xrefmaybe,frame,true);
259            Set(R, "$inscond", xref.inscond.value);
260            bool glob2 = xref.p_field->p_ctype == ns.c_globfld->p_ctype;
261            Set(R, "$thisparname", glob2 || !keyfld ? strptr() : strptr(tempstr() << "*p_"<<name_Get(*keyfld)));
262            if (upptr) {
263                Ins(&R, xrefmaybe.body    , "// $name: save pointer to $parent");
264            } else {
265                Ins(&R, xrefmaybe.body    , "// insert $name into index $xrefname");
266            }
267            Ins(&R, xrefmaybe.body    , "if ($inscond) { // user-defined insert condition");
268            // check if reftype has an InsertMaybe function that checks for duplicates
269            if (xref.p_field->reftype == dmmeta_Reftype_reftype_Thash || xref.p_field->reftype == dmmeta_Reftype_reftype_Ptr) {
270                Ins(&R, xrefmaybe.body, "    bool success = $xrefname_InsertMaybe($thisparname, row);");
271                Ins(&R, xrefmaybe.body, "    if (UNLIKELY(!success)) {");
272                Ins(&R, xrefmaybe.body, "        ch_RemoveAll(algo_lib::_db.errtext);");
273                Ins(&R, xrefmaybe.body, "        algo_lib::_db.errtext << \"$ns.duplicate_key  xref:$xrefkey\"; // check for duplicate key");
274                Ins(&R, xrefmaybe.body, "        return false;");
275                Ins(&R, xrefmaybe.body, "    }");
276            } else if (upptr) {
277                Ins(&R, xrefmaybe.body, "    row.$xrefname = p_$parent;");
278            } else {
279                Ins(&R, xrefmaybe.body, "    $xrefname_Insert($thisparname, row);");
280            }
281            Ins(&R, xrefmaybe.body    , "}");
282        }ind_end;
283    }
284    Ins(&R, xrefmaybe.body, "return retval;");
285}

387        Ins(&R, hash.body, "return prev;");

391// Comparison function for ctype
392// In amc, equality is not strictly a property of a type; it can be overriden on a per-field basis
393// For example, the same field (algo.Smallstr100 for instance) can be made case-insensitive

396void amc::tfunc_Ctype_Cmp() {
397    algo_lib::Replscope R;
398    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
399    if (ctype.c_ccmp) {
400        vrfy(!ctype.c_varlenfld,tempstr()<< "comparison function of varlength records is not supported ("<<ctype.ctype<<")");
401        Set(R, "$ByvalArgtype", ByvalArgtype(ctype));
402        Set(R, "$Cpptype", ctype.cpp_type);
403        Set(R, "$Ctype", ctype.ctype);
404        Set(R, "$Name", name_Get(ctype));
405
406        amc::FFunc& cmp = amc::CreateCurFunc();
407        Ins(&R, cmp.ret  , "i32", false);
408        Ins(&R, cmp.proto, "$Name_Cmp($ByvalArgtype lhs, $ByvalArgtype rhs)", false);
409        cmp.extrn = ctype.c_ccmp->extrn;
410        cmp.inl = c_datafld_N(ctype) < 4;
411        Ins(&R, cmp.body    , "i32 retval = 0;");
412        bool need_test = false;
413        ind_beg(amc::ctype_c_datafld_curs, field,ctype) {
414            tempstr test;
415            if (PadQ(field)) {
416            } else if (amc::ind_func_Find(dmmeta::Func_Concat_field_name(field.field,"Cmp"))) {// check field-spsecific compare
417                Set(R, "$name", name_Get(field));
418                Ins(&R, test    , "retval = $name_Cmp(lhs,rhs);");
419            } else if (field.reftype == dmmeta_Reftype_reftype_Upptr || field.reftype == dmmeta_Reftype_reftype_Ptr) {// default for pointers
420                Set(R, "$a_val", FieldvalExpr(&ctype, field, "lhs"));
421                Set(R, "$b_val", FieldvalExpr(&ctype, field, "rhs"));
422                Set(R, "$Fldtype", field.p_arg->cpp_type);
423                Ins(&R, test, "retval = u64_Cmp((u64)(void*)$a_val, (u64)(void*)$b_val);");
424            } else if (field.reftype == dmmeta_Reftype_reftype_Smallstr) {// default for small strings
425                Set(R, "$name", name_Get(field));
426                Ins(&R, test, "retval = algo::strptr_Cmp($name_Getary(lhs), $name_Getary(rhs));");
427            } else if (field.reftype == dmmeta_Reftype_reftype_Base) {
428                // compare bases???
429            } else if (field.p_arg->c_ccmp) {// type-specific compare
430                Set(R, "$a_val", FieldvalExpr(&ctype, field, "lhs"));
431                Set(R, "$b_val", FieldvalExpr(&ctype, field, "rhs"));
432                Set(R, "$Fldtype", field.p_arg->cpp_type);
433                Ins(&R, test, "retval = $Fldtype_Cmp($a_val, $b_val);");
434            } else {
435                prerr("amc.bad_cmp"
436                      <<Keyval("field",field.field)
437                      <<Keyval("offending_type",field.arg)
438                      <<Keyval("comment","No comparison function defined for offending type"));
439                algo_lib::_db.exit_code=1;
440            }
441            if (ch_N(test) > 0) {
442                if (need_test) {
443                    Ins(&R, cmp.body, "if (retval != 0) {");
444                    Ins(&R, cmp.body, "    return retval;");
445                    Ins(&R, cmp.body, "}");
446                }
447                cmp.body << test;
448                need_test = true;
449            }
450        }ind_end;
451        Ins(&R, cmp.body, "return retval;");
452    }

469        } else if (field.reftype == dmmeta_Reftype_reftype_Smallstr) {// default for small strings
470            Set(R, "$name", name_Get(field));
471            Ins(&R, oplt.body, "return algo::strptr_Lt($name_Getary(lhs), $name_Getary(rhs));");

480                  <<Keyval("offending_type",field.arg)

487// Less-than function for ctype
488void amc::tfunc_Ctype_Lt() {
489    algo_lib::Replscope R;
490    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
491
492    if (ctype.c_ccmp && ctype.c_ccmp->order) {
493        Set(R, "$ByvalArgtype", ByvalArgtype(ctype));
494        Set(R, "$Ctype", ctype.ctype);
495        Set(R, "$Name", name_Get(ctype));
496
497        amc::FFunc& oplt = amc::CreateCurFunc();
498        Ins(&R, oplt.ret  , "bool", false);
499        Ins(&R, oplt.proto, "$Name_Lt($ByvalArgtype lhs, $ByvalArgtype rhs)", false);
500        oplt.extrn = ctype.c_ccmp->extrn;
501        oplt.inl = true;
502        // Lt function is faster than Cmp because there are fewer calculations to do
503        // But if there is more than one field the advantage goes away because if a.x is not less
504        // than b.x, you would have to check whether a.x==b.x before proceeding to the next field
505        // So if there is more than one field, we revert to the generic Cmp.
506        if (c_datafld_N(ctype) == 1) { // special case -- single field.
507            Ctype_Lt_SingleField(R,ctype,oplt);
508        } else {
509            Ins(&R, oplt.body    , "return $Name_Cmp(lhs,rhs) < 0;");
510        }
511    }

643// Emit optimal padded string comparison code
644// to string OUT
645// The fields being compared are FIELD in "lhs" and "rhs"
646// The resulting value is written to "retval" which must be initially true.
647static bool GenSmallstrEq(algo_lib::Replscope &R, amc::FField &field, cstring &out) {
648    bool ret=false;
649    amc::FSmallstr *smallstr = field.c_smallstr;
650    // compare small strings bytewise
651    if (smallstr && (smallstr->strtype == dmmeta_Strtype_strtype_rightpad || smallstr->strtype == dmmeta_Strtype_strtype_leftpad)) {
652        int len = smallstr->length;
653        // for big strings
654        if (len>128) {
655            int niter=(len/8);
656            Set(R, "$niter", tempstr()<<niter);
657            Ins(&R, out, "for (int i=0; i<$niter && retval; i++) {");
658            Ins(&R, out, "    retval = ((u64*)lhs.$name)[i] == ((u64*)rhs.$name)[i];");
659            Ins(&R, out, "}");
660            len -= niter*8;
661        }
662        // remainder
663        if (len>0) {
664            Ins(&R, out, "retval = retval ",false);
665            while (len>0) {
666                Set(R, "$offset", tempstr()<< smallstr->length - len);
667                if (len>=8) {
668                    Ins(&R, out, "\n &&*(u64*)(lhs.$name+$offset) == *(u64*)(rhs.$name+$offset)",false);
669                    len-=8;
670                } else if (len >=4) {
671                    Ins(&R, out, "\n &&*(u32*)(lhs.$name+$offset) == *(u32*)(rhs.$name+$offset)",false);
672                    len-=4;
673                } else if (len >=2) {
674                    Ins(&R, out, "\n &&*(u16*)(lhs.$name+$offset) == *(u16*)(rhs.$name+$offset)",false);
675                    len-=2;
676                } else if (len >=1) {
677                    Ins(&R, out, "\n &&*(u8*)(lhs.$name+$offset) == *(u8*)(rhs.$name+$offset)",false);
678                    len--;
679                }
680            }
681            Ins(&R,out,";");
682        }
683        ret=true;
684    }
685    return ret;
686}

688// Equality function for ctype
689// The function proceeds field-by-field, using the following rules:
690// - pad bytes are ignored
691// - if a field has an Eq function defined (usually because its equality has been customized with the fcmp record)
692//   we call out to the defined Eq function.
693// - pointers and big-endian fields are compared bitwise for equality
694// - any string fields are compared as regular strings
695// - if no custom Eq function is defined on a field, but the field's type has an Eq function, it is used
696// - for all other fields, c++ operator "==" is used. it better be defined
697void amc::tfunc_Ctype_Eq() {
698    algo_lib::Replscope R;
699    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
700
701    if (ctype.c_ccmp) {
702        vrfy(!ctype.c_varlenfld,tempstr()<< "comparison function of varlength records is not supported ("<<ctype.ctype<<")");
703        Set(R, "$Cpptype", ctype.cpp_type);
704        Set(R, "$Ctype", ctype.ctype);
705        amc::FFunc& opeq = amc::CreateCurFunc(true);
706        AddRetval(opeq,"bool","retval","true");
707        AddProtoArg(opeq, ByvalArgtype(ctype,false), "lhs");
708        AddProtoArg(opeq, ByvalArgtype(ctype,false), "rhs");
709        opeq.extrn = ctype.c_ccmp->extrn;
710        opeq.inl = c_datafld_N(ctype) < 4;
711        bool need_test = false;
712        ind_beg(amc::ctype_c_datafld_curs, field,ctype) {
713            tempstr test;
714            if (PadQ(field)) {
715            } else if (amc::ind_func_Find(dmmeta::Func_Concat_field_name(field.field,"Eq"))) {
716                Set(R, "$name", name_Get(field));
717                Ins(&R, test, "retval = $name_Eq(lhs,rhs);");
718            } else if (field.reftype == dmmeta_Reftype_reftype_Upptr || field.reftype == dmmeta_Reftype_reftype_Ptr) {// default for pointers
719                Set(R, "$a_val", FieldvalExpr(&ctype, field, "lhs"));
720                Set(R, "$b_val", FieldvalExpr(&ctype, field, "rhs"));
721                Set(R, "$Fldtype", field.p_arg->cpp_type);
722                Ins(&R, test, "retval = u64_Eq((u64)(void*)$a_val, (u64)(void*)$b_val);");
723            } else if (field.reftype == dmmeta_Reftype_reftype_Smallstr) {// default for small strings
724                Set(R, "$name", name_Get(field));
725                if (!GenSmallstrEq(R, field, test)) {
726                    Ins(&R, test, "retval = algo::strptr_Eq($name_Getary(lhs), $name_Getary(rhs));");
727                }
728            } else if (field.reftype == dmmeta_Reftype_reftype_Base) {
729                // fields have already been imported, ignore
730            } else if (field.p_arg->c_ccmp) {// type-specific compare
731                Set(R, "$a_val", FieldvalExpr(&ctype, field, "lhs"));
732                Set(R, "$b_val", FieldvalExpr(&ctype, field, "rhs"));
733                Set(R, "$Fldtype", field.p_arg->cpp_type);
734                Ins(&R, test, "retval = $Fldtype_Eq($a_val, $b_val);");
735            } else {
736                Set(R, "$name", name_Get(field));
737                Set(R,"$suffix",field.c_fbigend?"_be":"");
738                // hack: bigend fields do not need swapping to compare
739                Ins(&R, test, "retval = lhs.$name$suffix==rhs.$name$suffix;");
740            }
741            if (ch_N(test) > 0) {
742                if (need_test) {
743                    Ins(&R, opeq.body, "if (!retval) {");
744                    Ins(&R, opeq.body, "    return false;");
745                    Ins(&R, opeq.body, "}");
746                }
747                opeq.body << test;
748                need_test = true;
749            }
750        }ind_end;
751    }
752}

889void amc::tfunc_Ctype_EqEnum() {
890    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
891    amc::FField *field = c_datafld_N(ctype) == 1 ? c_datafld_Find(ctype,0) : NULL;
892    // always generate EqEnum function if field has fconsts.
893    // TODO: generate EqEnum for field, then promote to a ctype-wide EqEnum
894    // if only one exists.
895    // NOTE: do not look for ccmp; comparison with enum is generated because the
896    // enum exists.
897    bool doit = field
898        && c_fconst_N(*field)
899        && !amc::FieldStringQ(*field)
900        && (ctype.c_ccmp && ctype.c_ccmp->genop);
901    if (doit) {
902        algo_lib::Replscope R;
903        Set(R, "$Get", FieldvalExpr(&ctype,*field,"lhs"));
904        Set(R, "$ByvalArgtype", amc::ByvalArgtype(ctype));
905        Set(R, "$Ctype", ctype.ctype);
906        Set(R, "$Name", name_Get(ctype));
907
908        Set(R, "$Enumtype", Enumtype(*c_datafld_Find(ctype,0)));
909        amc::FFunc& opeqenum = amc::CreateCurFunc();
910        opeqenum.comment = "define enum comparison operator to avoid ambiguity";
911        Ins(&R, opeqenum.ret  , "bool", false);
912        Ins(&R, opeqenum.proto, "$Name_EqEnum($ByvalArgtype lhs, $Enumtype rhs)", false);
913        opeqenum.glob = true;
914        opeqenum.inl = true;
915        // field type may be unsigned, in which case signed<>unsigned comparison
916        // warning fires. insert an extra cast to Enumtype to avoid it.
917        Ins(&R, opeqenum.body, "return $Enumtype($Get) == rhs;");
918    }
919}

923void amc::tfunc_Ctype_GetAnon() {
924    amc::FCtype &ctype = *amc::_db.genctx.p_ctype;
925    bool need_anon = false;
926    ind_beg(amc::ctype_zs_cfmt_curs,cfmt,ctype) if (cfmt.read
927                                                    && !(cfmt.printfmt == dmmeta_Printfmt_printfmt_Raw
928                                                         || cfmt.printfmt == dmmeta_Printfmt_printfmt_Extern)) {
929        need_anon = true;
930    }ind_end;
931
932    int n_anon = c_anonfld_N(ctype);
933    if (need_anon && n_anon > 0) {
934        algo_lib::Replscope R;
935        Set(R, "$Ctype", ctype.ctype);
936        Set(R, "$Name", name_Get(ctype));
937        Set(R, "$Cpptype", ctype.cpp_type);
938        amc::FFunc& getanon = amc::CreateCurFunc();
939        Ins(&R, getanon.ret  , "algo::strptr", false);
940        Ins(&R, getanon.proto, "$Name_GetAnon($Cpptype &parent, i32 idx)",false);
941        MaybeUnused(getanon, "parent");//must call before "return"
942        Ins(&R, getanon.body   , "switch(idx) {");
943        int idx=0;
944        Set(R,"$dfltanon","algo::strptr()");
945        ind_beg(amc::ctype_c_field_curs, field,ctype) if (field.c_anonfld) {
946            Set(R, "$fstr"   , strptr(tempstr() << "strptr(\"" << name_Get(field) << "\", "<<ch_N(name_Get(field))<<")"));
947            Set(R, "$idx"   , tempstr()<<idx);
948            // Tary must be last arg
949            if (field.reftype == dmmeta_Reftype_reftype_Tary) {
950                Set(R,"$dfltanon","$fstr");
951                break;
952            } else {
953                Ins(&R, getanon.body, "case($idx): return $fstr;");
954            }
955            idx++;
956        }ind_end;
957        Ins(&R, getanon.body    , "default: return $dfltanon;");
958        Ins(&R, getanon.body    , "}");
959    }
960}
