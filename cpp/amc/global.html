 312void amc::tfunc_Global_main() {
 313    algo_lib::Replscope &R = amc::_db.genctx.R;
 314    amc::FField &field = *amc::_db.genctx.p_field;
 315    amc::FNs &ns = *field.p_ctype->p_ns;
 316    if (amc::c_fstep_N(ns) > 0 && !ns.c_main && ExeQ(ns)) {
 317        prerr("amc.no_main"
 318              <<Keyval("ns",ns.ns)
 319              <<Keyval("comment","Add dmmeta.main record for this namespace"));
 320        algo_lib::_db.exit_code=1;
 321    }
 322    // generate main() function
 323    if (ns.c_main && ExeQ(ns)) {
 324        // main
 325        amc::FFunc& main = amc::ind_func_GetOrCreate(Subst(R, "$ns...main"));
 326        main.glob = true;
 327        main.globns = true;
 328        main.ret = "int";
 329        Ins(&R, main.proto, "main(int argc, char **argv)", false);
 330        Ins(&R, main.body    , "try {");
 331        ind_beg(amc::ns_c_parentns_curs, parentns, ns) if (parentns.c_globfld) {
 332            Ins(&R, main.body, tempstr()<<"    "<<parentns.c_globfld->p_arg->cpp_type<<"_Init();");
 333        }ind_end;
 334        Ins(&R, main.body    , "    algo_lib::_db.argc = argc;");
 335        Ins(&R, main.body    , "    algo_lib::_db.argv = argv;");
 336        Ins(&R, main.body    , "    algo_lib::IohookInit();");
 337
 338        // call last non-module main, followed by all module mains
 339        // #AL# this must be cleaned up -- only used for lib_m2
 340        for (int i=c_parentns_N(ns)-1; i>=0; i--) {// call first non-module main
 341            amc::FNs &parentns=*c_parentns_Find(ns,i);
 342            if (parentns.c_main && !parentns.c_main->ismodule) {
 343                Set(R, "$parentns", parentns.ns);
 344                if (parentns.c_fcmdline && parentns.c_fcmdline->read) {
 345                    Ins(&R, main.body, "$parentns::ReadArgv(); // dmmeta.main:$parentns");
 346                }
 347                Ins(&R, main.body, "$parentns::Main(); // user-defined main");
 348                break;
 349            }
 350        }
 351
 352        ind_beg(amc::ns_c_parentns_curs,parentns,ns) if (parentns.c_main && parentns.c_main->ismodule) {// call all ismodule mains
 353            Set(R, "$parentns", parentns.ns);
 354            if (parentns.c_fcmdline && parentns.c_fcmdline->read) {
 355                Ins(&R, main.body, "$parentns::ReadArgv(); // dmmeta.main:$parentns  ismodule:Y");
 356            }
 357            Ins(&R, main.body, "$parentns::Main(); // call through to user-defined main");
 358        }ind_end;
 359
 360        for (int i=c_parentns_N(ns)-1; i>=0; i--) {// call main loop of ismodule main
 361            amc::FNs &parentns=*c_parentns_Find(ns,i);
 362            if (parentns.c_main && parentns.c_main->ismodule) {
 363                Set(R, "$parentns", parentns.ns);
 364                Ins(&R, main.body    , "    $parentns::MainLoop(); // dmmeta.main:$parentns  ismodule:Y");
 365                break;
 366            }
 367        }
 368        Ins(&R, main.body    , "} catch(algo_lib::ErrorX &x) {");
 369        Ins(&R, main.body    , "    prerr(\"$ns.error  \" << x); // there may be additional hints in DetachBadTags");
 370        Ins(&R, main.body    , "    algo_lib::_db.exit_code = 1;");

 499// print rhs as a multiline c++ string
 500// Example:
 501// FmtCppStr(lhs, "ab\tcd\nefg");
 502// output:
 503// "ab\tcd\n"
 504// "efg"
 505// Note in the above example '\t' was converted to a tab by c++ compiler,
 506// and escaped again by FmtCppStr.
 507static void FmtCppStr(cstring &lhs, strptr rhs) {
 508    int start = 0;
 509    do {
 510        int end = FindFrom(rhs, '\n', start);
 511        if (end==-1) end=rhs.n_elems; else end++;
 512        strptr_PrintCpp(GetRegion(rhs,start,end-start), lhs);
 513        start=end;
 514        lhs << eol;
 515    } while (start < rhs.n_elems);
 516    lhs << ";\n";
 517}

 519// Generate help strings for command-line tools.
 520//
 521// Example Output:
 522//
 523// const char *dmsess_help =
 524// "Usage: dmsess [sess] [options]\n"
 525// "  sess          session name\n"
 526// "  -list         list processes in session"
 527// "  -cfg:string   (with -create) debug or release. default: debug"
 528// ;
 529//
 530// generate help string
 531// "Usage: <stripped name> <non-optional args> <options>\n"
 532// <non-optional args>: any arg without default value (whether named or not), excluding bool
 533// <options> true if any options defined
 534// <table of non-optional args>
 535// Options
 536// <list of optional args>
 537// table: compute width or largest name
 538// prepend - to field names which are not bracketed.
 539// append ". default: XXXX" to options which have a default
 540//
 541
 542static tempstr GetCmdArgType(amc::FField& field) {
 543    tempstr ret("string");// default
 544    amc::FArgvtype *argvtype = field.p_arg->c_argvtype;
 545    // amc rewrites all pkey fields so they become Vals
 546    // so this first line can never execute
 547    if (c_fconst_N(*GetEnumField(field))) {
 548        ret = "enum";
 549    } else if (field.c_fflag) {
 550        ret = "flag";
 551    } else if (field.reftype == dmmeta_Reftype_reftype_Pkey) {
 552        ret = "pkey";
 553    } if (field.reftype == dmmeta_Reftype_reftype_Regx) {
 554        ret = "regx";
 555    } else {
 556        if (!argvtype && c_field_N(*field.p_arg) == 1) {
 557            argvtype = c_field_Find(*field.p_arg,0)->p_ctype->c_argvtype;
 558        }
 559        if (argvtype) {
 560            ret=argvtype->argvtype;
 561        }
 562    }
 563    return ret;
 564}

 598// Adjust displayed default
 599// Translate true/false into "Y"/<empty string>
 600static tempstr GetCmdArgDflt(amc::FField &field) {
 601    tempstr ret(field.dflt.value);
 602
 603    if (field.c_fflag) {
 604        // no default for flags (even if they can take a value)
 605        ret="";
 606    } if (field.c_tary) {
 607        // no default for arrays -- since they can be empty
 608        ret="";
 609    } else if (field.arg=="bool") {
 610        if (field.dflt.value=="true") {
 611            ret="Y";
 612        } else if (field.dflt.value=="false") {
 613            ret="";// do not show
 614        }
 615    } else {
 616        // scan enums for the field and translate back from the
 617        // numeric value to the symbolic
 618        tempstr enum_dflt;
 619        // scan enums...
 620        ind_beg(amc::field_c_fconst_curs,fconst,*GetEnumField(field)) {
 621            if (fconst.value.value == field.dflt.value
 622                || (!ch_N(field.dflt.value) && (fconst.value.value=="0" || fconst.value.value=="'\\0'"))) {
 623                enum_dflt = name_Get(fconst);
 624            }
 625        }ind_end;
 626        if (ch_N(enum_dflt) && enum_dflt != "\"\"") {
 627            ret = enum_dflt;
 628        }
 629    }
 630    return ret;
 631}

 635tempstr amc::GetUsageString(amc::FNs &ns, amc::FFcmdline &cmdline) {
 636    amc::FCtype &ctype = *cmdline.p_field->p_arg;
 637    tempstr cmd_format(name_Get(ctype));
 638    bool has_opts = false;
 639    ind_beg(amc::ctype_c_field_curs, field, ctype) if (!field.c_falias) {
 640        if (CmdArgRequiredQ(field)) {
 641            if (CmdArgValueRequiredQ(field)) {
 642                if (field.c_anonfld) {
 643                    // [-str:]<string>   -- CmdArgRequired, anon, value required
 644                    cmd_format << " [-"<<name_Get(field)<<":]<"<<GetCmdArgType(field)<<">";
 645                } else {
 646                    // -astr:<string>    -- CmdArgRequired, named, value required
 647                    cmd_format << " -"<<name_Get(field)<<":<"<<GetCmdArgType(field)<<">";
 648                }
 649            }
 650        } else {
 651            if (CmdArgValueRequiredQ(field)) {
 652                if (field.c_anonfld) {
 653                    // [[-str:]<string>] -- !CmdArgRequired, anon, value required
 654                    cmd_format << " [[-"<<name_Get(field)<<":]<"<<GetCmdArgType(field)<<">]";
 655                } else {
 656                    //                   -- !CmdArgRequired, named, value not required -- skipped
 657                    has_opts = true;
 658                }
 659            }
 660        }
 661    }ind_end;
 662    if (has_opts) {
 663        cmd_format << " [options]";
 664    }
 665    tempstr ret;
 666    if (ch_N(ns.comment.value)) {
 667        ret << name_Get(ctype) << ": "<<ns.comment<<eol;
 668    }
 669    ret << "Usage: "<< cmd_format<<eol;
 670    return ret;
 671}

 675// CTYPE: type of command line
 676// NS: target namespace (actual executable)
 677static void GenHelpSyntax(amc::FNs &ns, amc::FFcmdline &cmdline) {
 678    amc::FCtype &ctype = *cmdline.p_field->p_arg;
 679    tempstr name(name_Get(ctype));
 680    // help message
 681    *ns.hdr << "extern const char *"<<name<<"_help;" << eol;
 682    *ns.cpp << "namespace "<<ns.ns<<" {"<<eol;
 683    *ns.cpp << "const char *"<<name<<"_help =" << eol;
 684    tempstr table;
 685    tempstr usage = GetUsageString(ns,cmdline);
 686    table << "    OPTION\tTYPE\tDFLT\tCOMMENT\n";
 687    // loop over fields of this command line, and its base command line
 688    for (int i=0; i<2; i++) {
 689        amc::FCtype *thisctype=i==0 ? &ctype : cmdline.p_basecmdline->p_arg;
 690        if (thisctype) {
 691            ind_beg(amc::ctype_c_field_curs, field, *thisctype) if (!field.c_falias) {
 692                table << "    ";
 693                table << (field.c_anonfld ? "" : "-");
 694                table << CmdargName(&field);
 695                table << "\t";
 696                table << GetCmdArgType(field);
 697                table << "\t";
 698                table << GetCmdArgDflt(field);
 699                table << "\t";
 700                table << field.comment;
 701                // list aliases
 702                ind_beg(amc::ctype_c_field_curs, aliasfield, *thisctype) {
 703                    if (aliasfield.c_falias && aliasfield.c_falias->p_srcfield == &field) {
 704                        table << "; alias -"<<name_Get(aliasfield);
 705                    }
 706                }ind_end;
 707                // explain cumulative flag
 708                if (field.c_fflag && field.c_fflag->cumulative) {
 709                    table << "; cumulative";
 710                }
 711                // provide single-line enum choice
 712                // and explain enum values (with comments)
 713                tempstr fconst_values;
 714                tempstr fconst_choice;
 715                algo::ListSep ls("|");
 716                if (!field.c_tary) {
 717                    ind_beg(amc::field_c_fconst_curs,fconst,*GetEnumField(field)) {
 718                        fconst_choice << ls << name_Get(fconst);
 719                        if (ch_N(fconst.comment)) {
 720                            fconst_values << "\t\t\t    " << name_Get(fconst) << "  " << fconst.comment << eol;
 721                        }
 722                    }ind_end;
 723                }
 724                // fconst choice
 725                if (ch_N(fconst_choice)) {
 726                    table << " (" << fconst_choice << ")";
 727                }
 728                table << eol;
 729                table << fconst_values; // fconst values
 730            }ind_end;
 731        }
 732    }
 733    usage << Tabulated(table, "\t", "lll",2);
 734
 735    tempstr lc;
 736    lc << name;
 737    MakeLower(lc);
 738    FmtCppStr(*ns.cpp, usage);
 739    *ns.cpp << eol << eol;
 740    *ns.cpp << "} // namespace "<<ns.ns << eol;
 741}

 865        Ins(&R, func.body, "                attrname=ch_FirstN(attrname,colon.beg);");
 866        Ins(&R, func.body, "                haveval=true;");
 867        Ins(&R, func.body, "            }");
 868        Ins(&R, func.body, "            // look up which command (this one or the base) contains the field");
 869        Ins(&R, func.body, "            whichns=0;");
 870        Ins(&R, func.body, "            needarg=-1;");
 871        if (basecmdline) {
 872            Ins(&R, func.body, "        // look up parameter information in base namespace (needarg will be -1 if lookup fails)");

 874            Ins(&R, func.body, "            needarg = $basecmdlinetypens::$basecmdlinectypename_NArgs(baseattrid,dfltval,&isanon);");
 875            Ins(&R, func.body, "        }");
 876        }
 877        Ins(&R, func.body, "            if (needarg<0) {");
 878        Ins(&R, func.body, "                whichns=1;");
 879        Ins(&R, func.body, "                // look up parameter information in this namespace (needarg will be -1 if lookup fails)");
 880        Ins(&R, func.body, "                if ($cmdlinetypens::FieldId_ReadStrptrMaybe(attrid,attrname)) {");
 881        Ins(&R, func.body, "                    needarg = $cmdlinetypens::$cmdlinectypename_NArgs(attrid,dfltval,&isanon);");
 882        Ins(&R, func.body, "                }");
 883        Ins(&R, func.body, "            }");
 884        Ins(&R, func.body, "            if (attrval == \"\" && dfltval != \"\") {");
 885        Ins(&R, func.body, "                attrval=dfltval;");
 886        Ins(&R, func.body, "                haveval=true;");
 887        Ins(&R, func.body, "            }");
 888        Ins(&R, func.body, "            if (needarg<0) {");
 889        Ins(&R, func.body, "                err<<\"$ns: unknown option \"<<Keyval(\"value\",arg)<<eol;");
 890        Ins(&R, func.body, "            } else {");
 891        if (n_anon>0) {
 892            Ins(&R, func.body, "            if (isanon) {");
 893            Ins(&R, func.body, "                if (attrname == nextanon) { // treat named anon (positional) argument as unnamed");
 894            Ins(&R, func.body, "                    attrname = \"\"; // treat it as unnamed");
 895            Ins(&R, func.body, "                } else if (nextanon != \"\") { // disallow out-of-order anon (positional) args");
 896            Ins(&R, func.body, "                    err<<\"$ns: error at \"<<algo::strptr_ToSsim(arg)<<\": must be preceded by [-\"<<nextanon<<\"]\"<<eol;");
 897            Ins(&R, func.body, "                }");
 898            Ins(&R, func.body, "            }");
 899        }
 900        Ins(&R, func.body, "            }");
 901        Ins(&R, func.body, "        }");
 902        if (n_anon>0) {
 903            Ins(&R, func.body, "        // look up anon field name based on index");
 904            Ins(&R, func.body, "        // anon fields are only allowed in the leaf ns, never base");
 905            Ins(&R, func.body, "        if (ch_N(attrname) == 0) {");
 906            Ins(&R, func.body, "            attrname = nextanon;");
 907            Ins(&R, func.body, "            nextanon = command::$ns_GetAnon(cmd, ++anonidx);");
 908            Ins(&R, func.body, "            $cmdlinetypens::FieldId_ReadStrptrMaybe(attrid,attrname);");
 909            Ins(&R, func.body, "            whichns=1;");
 910            Ins(&R, func.body, "        }");
 911        }
 912        Ins(&R, func.body, "        if (ch_N(attrname) == 0) {");
 913        Ins(&R, func.body, "            err << \"$ns: too many arguments. error at \"<<algo::strptr_ToSsim(arg)<<eol;");
 914        Ins(&R, func.body, "        }");
 915        Ins(&R, func.body, "        // read value into currently selected arg");
 916        Ins(&R, func.body, "        if (haveval) {");
 917        Ins(&R, func.body, "            bool ret=false;");
 918        Ins(&R, func.body, "            // it's already known which namespace is consuming the args,");
 919        Ins(&R, func.body, "            // so directly go there");
 920        if (basecmdline) {
 921            Ins(&R, func.body, "        if (whichns == 0) {");
 922            Ins(&R, func.body, "            ret=$basecmdlinetypens::$basecmdlinectypename_ReadFieldMaybe(base, attrname, attrval);");
 923            Ins(&R, func.body, "        }");
 924        }
 925        // mark required fields as present
 926        Ins(&R, func.body, "            if (whichns==1) {");
 927        Ins(&R, func.body, "                ret=$cmdlinetypens::$cmdlinectypename_ReadFieldMaybe(cmd, attrname, attrval);");
 928        Ins(&R, func.body, "                switch(attrid.value) {");
 929        ind_beg(ctype_c_field_curs,reqfield,*cmdline_ctype) if (CmdArgRequiredQ(reqfield)) {
 930            Set(R,"$reqfieldname",name_Get(reqfield));
 931            Ins(&R, func.body, "                case $cmdlinetypens_FieldId_$reqfieldname: $reqfieldname_present=true; break;");

 933        Ins(&R, func.body, "                    default:break;");
 934        Ins(&R, func.body, "                }");
 935        Ins(&R, func.body, "            }");
 936
 937        Ins(&R, func.body, "            if (!ret) {");
 938        Ins(&R, func.body, "                err<<\"$ns: error in \"");
 939        Ins(&R, func.body, "                    <<Keyval(\"option\",attrname)");
 940        Ins(&R, func.body, "                    <<Keyval(\"value\",attrval)<<eol;");
 941        Ins(&R, func.body, "            }");
 942        Ins(&R, func.body, "            needarg--;");
 943        Ins(&R, func.body, "            if (needarg <= 0) {");
 944        Ins(&R, func.body, "                attrname=\"\";// forget which argument was being filled");
 945        Ins(&R, func.body, "            }");
 946        Ins(&R, func.body, "        }");
 947        Ins(&R, func.body, "    }");

 949        Ins(&R, func.body, "bool dohelp = false;");

 955            Ins(&R, func.body, "if (algo_lib::_db.cmdline.help) {");
 956            Ins(&R, func.body, "    dohelp = true;");
 957            Ins(&R, func.body, "    doexit = true;");
 958            Ins(&R, func.body, "    algo_lib::_db.exit_code = 1; // help exits with non-zero status code");

 960            Ins(&R, func.body, "    // -ver output goes to stdout");

 965            Ins(&R, func.body, "        // dispsig goes to stdout");
 966            Ins(&R, func.body, "        dmmeta::Dispsigcheck out;");
 967            Ins(&R, func.body, "        dispsigcheck_CopyOut(dispsigcheck,out);");
 968            Ins(&R, func.body, "        prlog(out);");
 969            Ins(&R, func.body, "    }ind_end");
 970            Ins(&R, func.body, "    doexit = true;");
 971            Ins(&R, func.body, "}");
 972        }
 973
 974        Ins(&R, func.body, "if (!dohelp) {");
 975        ind_beg(ctype_c_field_curs,reqfield,*cmdline_ctype) if (CmdArgRequiredQ(reqfield)) {
 976            Set(R,"$reqfieldname",name_Get(reqfield));
 977            Ins(&R, func.body, "if (!$reqfieldname_present) {");
 978            Ins(&R, func.body, "    err << \"$ns: Missing value for required argument -$reqfieldname (see -help)\" << eol;");
 979            Ins(&R, func.body, "    doexit = true;");
 980            Ins(&R, func.body, "}");
 981        }ind_end;
 982        Ins(&R, func.body, "}");
 983
 984
 985        // post-processing steps:
 986        amc::FFloadtuples *floadtuples = fcmdline->p_field->p_arg->c_floadtuples;
 987        if (floadtuples) {
 988            Ins(&R, func.body, "// dmmeta.floadtuples:$cmdlinefield");
 989            Ins(&R, func.body, "if (!dohelp && err==\"\") {");
 990            Set(R,"$loadtuplesname",name_Get(*floadtuples->p_field));
 991            Ins(&R, func.body, "    algo_lib::ResetErrtext();");

 995            Ins(&R, func.body, "}");

 997
 998        Ins(&R, func.body, "if (err != \"\") {");
 999        Ins(&R, func.body, "    algo_lib::_db.exit_code=1;");
1000        Ins(&R, func.body, "    prerr(err);");
1001        Ins(&R, func.body, "    doexit=true;");
1002        Ins(&R, func.body, "}");
1003
1004        Ins(&R, func.body, "if (dohelp) {");
1005        Ins(&R, func.body, "    prlog($cmdlinectypename_help);");
1006        Ins(&R, func.body, "}");

1008        Ins(&R, func.body, "if (doexit) {");
