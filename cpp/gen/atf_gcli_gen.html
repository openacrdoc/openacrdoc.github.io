 673// --- atf_gcli.FDb.ind_gtblacttst.InsertMaybe
 674// Insert row into hash table. Return true if row is reachable through the hash after the function completes.
 675bool atf_gcli::ind_gtblacttst_InsertMaybe(atf_gcli::FGtblacttst& row) {
 676    ind_gtblacttst_Reserve(1);
 677    bool retval = true; // if already in hash, InsertMaybe returns true
 678    if (LIKELY(row.ind_gtblacttst_next == (atf_gcli::FGtblacttst*)-1)) {// check if in hash already
 679        u32 index = algo::Smallstr250_Hash(0, row.gtblacttst) & (_db.ind_gtblacttst_buckets_n - 1);
 680        atf_gcli::FGtblacttst* *prev = &_db.ind_gtblacttst_buckets_elems[index];
 681        do {
 682            atf_gcli::FGtblacttst* ret = *prev;
 683            if (!ret) { // exit condition 1: reached the end of the list
 684                break;
 685            }
 686            if ((*ret).gtblacttst == row.gtblacttst) { // exit condition 2: found matching key
 687                retval = false;
 688                break;
 689            }
 690            prev = &ret->ind_gtblacttst_next;
 691        } while (true);
 692        if (retval) {
 693            row.ind_gtblacttst_next = *prev;
 694            _db.ind_gtblacttst_n++;
 695            *prev = &row;
 696        }
 697    }
 698    return retval;
 699}

 701// --- atf_gcli.FDb.ind_gtblacttst.Remove
 702// Remove reference to element from hash index. If element is not in hash, do nothing
 703void atf_gcli::ind_gtblacttst_Remove(atf_gcli::FGtblacttst& row) {
 704    if (LIKELY(row.ind_gtblacttst_next != (atf_gcli::FGtblacttst*)-1)) {// check if in hash already
 705        u32 index = algo::Smallstr250_Hash(0, row.gtblacttst) & (_db.ind_gtblacttst_buckets_n - 1);
 706        atf_gcli::FGtblacttst* *prev = &_db.ind_gtblacttst_buckets_elems[index]; // addr of pointer to current element
 707        while (atf_gcli::FGtblacttst *next = *prev) {                          // scan the collision chain for our element
 708            if (next == &row) {        // found it?
 709                *prev = next->ind_gtblacttst_next; // unlink (singly linked list)
 710                _db.ind_gtblacttst_n--;
 711                row.ind_gtblacttst_next = (atf_gcli::FGtblacttst*)-1;// not-in-hash

 715        }

 719// --- atf_gcli.FDb.ind_gtblacttst.Reserve
 720// Reserve enough room in the hash for N more elements. Return success code.
 721void atf_gcli::ind_gtblacttst_Reserve(int n) {
 722    u32 old_nbuckets = _db.ind_gtblacttst_buckets_n;
 723    u32 new_nelems   = _db.ind_gtblacttst_n + n;
 724    // # of elements has to be roughly equal to the number of buckets
 725    if (new_nelems > old_nbuckets) {
 726        int new_nbuckets = i32_Max(algo::BumpToPow2(new_nelems), u32(4));
 727        u32 old_size = old_nbuckets * sizeof(atf_gcli::FGtblacttst*);
 728        u32 new_size = new_nbuckets * sizeof(atf_gcli::FGtblacttst*);
 729        // allocate new array. we don't use Realloc since copying is not needed and factor of 2 probably
 730        // means new memory will have to be allocated anyway
 731        atf_gcli::FGtblacttst* *new_buckets = (atf_gcli::FGtblacttst**)algo_lib::malloc_AllocMem(new_size);
 732        if (UNLIKELY(!new_buckets)) {
 733            FatalErrorExit("atf_gcli.out_of_memory  field:atf_gcli.FDb.ind_gtblacttst");
 734        }
 735        memset(new_buckets, 0, new_size); // clear pointers
 736        // rehash all entries
 737        for (int i = 0; i < _db.ind_gtblacttst_buckets_n; i++) {
 738            atf_gcli::FGtblacttst* elem = _db.ind_gtblacttst_buckets_elems[i];
 739            while (elem) {
 740                atf_gcli::FGtblacttst &row        = *elem;
 741                atf_gcli::FGtblacttst* next       = row.ind_gtblacttst_next;
 742                u32 index          = algo::Smallstr250_Hash(0, row.gtblacttst) & (new_nbuckets-1);
 743                row.ind_gtblacttst_next     = new_buckets[index];
 744                new_buckets[index] = &row;
 745                elem               = next;
 746            }
 747        }
 748        // free old array
 749        algo_lib::malloc_FreeMem(_db.ind_gtblacttst_buckets_elems, old_size);
 750        _db.ind_gtblacttst_buckets_elems = new_buckets;
 751        _db.ind_gtblacttst_buckets_n = new_nbuckets;
 752    }
 753}

 886// --- atf_gcli.FDb.c_gtblacttst.Remove
 887// Find element using linear scan. If element is in array, remove, otherwise do nothing
 888void atf_gcli::c_gtblacttst_Remove(atf_gcli::FGtblacttst& row) {
 889    if (bool_Update(row._db_c_gtblacttst_in_ary,false)) {
 890        int lim = _db.c_gtblacttst_n;
 891        atf_gcli::FGtblacttst* *elems = _db.c_gtblacttst_elems;
 892        // search backward, so that most recently added element is found first.
 893        // if found, shift array.
 894        for (int i = lim-1; i>=0; i--) {
 895            atf_gcli::FGtblacttst* elem = elems[i]; // fetch element
 896            if (elem == &row) {
 897                int j = i + 1;
 898                size_t nbytes = sizeof(atf_gcli::FGtblacttst*) * (lim - j);
 899                memmove(elems + i, elems + j, nbytes);
 900                _db.c_gtblacttst_n = lim - 1;
 901                break;
 902            }
 903        }
 904    }
 905}

1233// --- atf_gcli.FDb.ind_gclienv.Reserve
1234// Reserve enough room in the hash for N more elements. Return success code.
1235void atf_gcli::ind_gclienv_Reserve(int n) {
1236    u32 old_nbuckets = _db.ind_gclienv_buckets_n;
1237    u32 new_nelems   = _db.ind_gclienv_n + n;
1238    // # of elements has to be roughly equal to the number of buckets
1239    if (new_nelems > old_nbuckets) {
1240        int new_nbuckets = i32_Max(algo::BumpToPow2(new_nelems), u32(4));
1241        u32 old_size = old_nbuckets * sizeof(atf_gcli::FGclienv*);
1242        u32 new_size = new_nbuckets * sizeof(atf_gcli::FGclienv*);
1243        // allocate new array. we don't use Realloc since copying is not needed and factor of 2 probably
1244        // means new memory will have to be allocated anyway
1245        atf_gcli::FGclienv* *new_buckets = (atf_gcli::FGclienv**)algo_lib::malloc_AllocMem(new_size);
1246        if (UNLIKELY(!new_buckets)) {
1247            FatalErrorExit("atf_gcli.out_of_memory  field:atf_gcli.FDb.ind_gclienv");
1248        }
1249        memset(new_buckets, 0, new_size); // clear pointers
1250        // rehash all entries
1251        for (int i = 0; i < _db.ind_gclienv_buckets_n; i++) {
1252            atf_gcli::FGclienv* elem = _db.ind_gclienv_buckets_elems[i];
1253            while (elem) {
1254                atf_gcli::FGclienv &row        = *elem;
1255                atf_gcli::FGclienv* next       = row.ind_gclienv_next;
1256                u32 index          = algo::Smallstr50_Hash(0, row.gclienv) & (new_nbuckets-1);
1257                row.ind_gclienv_next     = new_buckets[index];
1258                new_buckets[index] = &row;
1259                elem               = next;
1260            }
1261        }
1262        // free old array
1263        algo_lib::malloc_FreeMem(_db.ind_gclienv_buckets_elems, old_size);
1264        _db.ind_gclienv_buckets_elems = new_buckets;
1265        _db.ind_gclienv_buckets_n = new_nbuckets;
1266    }
1267}

1410// Allocate space for one element. If no memory available, return NULL.

1413    // compute level and index on level
1414    u64 bsr   = algo::u64_BitScanReverse(new_nelems);
1415    u64 base  = u64(1)<<bsr;
1416    u64 index = new_nelems-base;
1417    void *ret = NULL;
1418    // if level doesn't exist yet, create it
1419    atf_gcli::FGtblact*  lev   = NULL;
1420    if (bsr < 32) {
1421        lev = _db.gtblact_lary[bsr];
1422        if (!lev) {
1423            lev=(atf_gcli::FGtblact*)algo_lib::malloc_AllocMem(sizeof(atf_gcli::FGtblact) * (u64(1)<<bsr));
1424            _db.gtblact_lary[bsr] = lev;
1425        }
1426    }
1427    // allocate element from this level
1428    if (lev) {
1429        _db.gtblact_n = i32(new_nelems);
1430        ret = lev + index;
1431    }
1432    return ret;

1801// Find element using linear scan. If element is in array, remove, otherwise do nothing
1802void atf_gcli::c_gclienvsub_Remove(atf_gcli::FGclienv& gclienv, atf_gcli::FGclienvsub& row) {
1803    if (bool_Update(row.gclienv_c_gclienvsub_in_ary,false)) {
1804        int lim = gclienv.c_gclienvsub_n;
1805        atf_gcli::FGclienvsub* *elems = gclienv.c_gclienvsub_elems;
1806        // search backward, so that most recently added element is found first.
1807        // if found, shift array.
1808        for (int i = lim-1; i>=0; i--) {
1809            atf_gcli::FGclienvsub* elem = elems[i]; // fetch element
1810            if (elem == &row) {
1811                int j = i + 1;
1812                size_t nbytes = sizeof(atf_gcli::FGclienvsub*) * (lim - j);
1813                memmove(elems + i, elems + j, nbytes);
1814                gclienv.c_gclienvsub_n = lim - 1;
1815                break;
1816            }
1817        }
1818    }
1819}
