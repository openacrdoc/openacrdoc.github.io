 491        if (reserved == 0) {
 492            break;
 493        }
 494    }
 495    return ret;
 496}

 498// --- atf_netio.FDb.endpoint.ReserveMem
 499// Allocate block of given size, break up into small elements and append to free list.
 500// Return number of elements reserved.
 501u64 atf_netio::endpoint_ReserveMem(u64 size) {
 502    u64 ret = 0;
 503    if (size >= sizeof(atf_netio::FEndpoint)) {
 504        atf_netio::FEndpoint *mem = (atf_netio::FEndpoint*)algo_lib::malloc_AllocMem(size);
 505        ret = mem ? size / sizeof(atf_netio::FEndpoint) : 0;
 506        // add newly allocated elements to the free list;
 507        for (u64 i=0; i < ret; i++) {
 508            mem[i].endpoint_next = _db.endpoint_free;
 509            _db.endpoint_free = mem+i;
 510        }
 511    }
 512    return ret;
 513}

 515// --- atf_netio.FDb.endpoint.XrefMaybe
 516// Insert row into all appropriate indices. If error occurs, store error
 517// in algo_lib::_db.errtext and return false. Caller must Delete or Unref such row.
 518bool atf_netio::endpoint_XrefMaybe(atf_netio::FEndpoint &row) {
 519    bool retval = true;
 520    (void)row;
 521    // insert endpoint into index zd_endpoint
 522    if (true) { // user-defined insert condition
 523        zd_endpoint_Insert(row);
 524    }
 525    // insert endpoint into index ind_endpoint
 526    if (true) { // user-defined insert condition
 527        bool success = ind_endpoint_InsertMaybe(row);
 528        if (UNLIKELY(!success)) {
 529            ch_RemoveAll(algo_lib::_db.errtext);
 530            algo_lib::_db.errtext << "atf_netio.duplicate_key  xref:atf_netio.FDb.ind_endpoint"; // check for duplicate key
 531            return false;
 532        }
 533    }
 534    return retval;
 535}

 537// --- atf_netio.FDb.zd_endpoint.Insert
 538// Insert row into linked list. If row is already in linked list, do nothing.
 539void atf_netio::zd_endpoint_Insert(atf_netio::FEndpoint& row) {
 540    if (!zd_endpoint_InLlistQ(row)) {
 541        atf_netio::FEndpoint* old_tail = _db.zd_endpoint_tail;
 542        row.zd_endpoint_next = NULL;
 543        row.zd_endpoint_prev = old_tail;
 544        _db.zd_endpoint_tail = &row;
 545        atf_netio::FEndpoint **new_row_a = &old_tail->zd_endpoint_next;
 546        atf_netio::FEndpoint **new_row_b = &_db.zd_endpoint_head;
 547        atf_netio::FEndpoint **new_row = old_tail ? new_row_a : new_row_b;
 548        *new_row = &row;
 549        _db.zd_endpoint_n++;
 550    }
 551}

 553// --- atf_netio.FDb.zd_endpoint.Remove
 554// Remove element from index. If element is not in index, do nothing.
 555void atf_netio::zd_endpoint_Remove(atf_netio::FEndpoint& row) {
 556    if (zd_endpoint_InLlistQ(row)) {
 557        atf_netio::FEndpoint* old_head       = _db.zd_endpoint_head;
 558        (void)old_head; // in case it's not used
 559        atf_netio::FEndpoint* prev = row.zd_endpoint_prev;
 560        atf_netio::FEndpoint* next = row.zd_endpoint_next;
 561        // if element is first, adjust list head; otherwise, adjust previous element's next
 562        atf_netio::FEndpoint **new_next_a = &prev->zd_endpoint_next;
 563        atf_netio::FEndpoint **new_next_b = &_db.zd_endpoint_head;
 564        atf_netio::FEndpoint **new_next = prev ? new_next_a : new_next_b;
 565        *new_next = next;
 566        // if element is last, adjust list tail; otherwise, adjust next element's prev
 567        atf_netio::FEndpoint **new_prev_a = &next->zd_endpoint_prev;
 568        atf_netio::FEndpoint **new_prev_b = &_db.zd_endpoint_tail;
 569        atf_netio::FEndpoint **new_prev = next ? new_prev_a : new_prev_b;
 570        *new_prev = prev;
 571        _db.zd_endpoint_n--;
 572        row.zd_endpoint_next=(atf_netio::FEndpoint*)-1; // not-in-list
 573    }
 574}

 609// --- atf_netio.FDb.ind_endpoint.Find
 610// Find row by key. Return NULL if not found.
 611atf_netio::FEndpoint* atf_netio::ind_endpoint_Find(const algo::strptr& key) {
 612    u32 index = algo::Smallstr50_Hash(0, key) & (_db.ind_endpoint_buckets_n - 1);
 613    atf_netio::FEndpoint* *e = &_db.ind_endpoint_buckets_elems[index];
 614    atf_netio::FEndpoint* ret=NULL;
 615    do {
 616        ret       = *e;
 617        bool done = !ret || (*ret).endpoint == key;
 618        if (done) break;
 619        e         = &ret->ind_endpoint_next;
 620    } while (true);
 621    return ret;
 622}

 633// Find row by key. If not found, create and x-reference a new row with with this key.
 634atf_netio::FEndpoint& atf_netio::ind_endpoint_GetOrCreate(const algo::strptr& key) {

 643        }

 645    vrfy(ret, tempstr() << "atf_netio.create_error  table:ind_endpoint  key:'"<<key<<"'  comment:'bad xref'");
 646    return *ret;
 647}

 649// --- atf_netio.FDb.ind_endpoint.InsertMaybe
 650// Insert row into hash table. Return true if row is reachable through the hash after the function completes.
 651bool atf_netio::ind_endpoint_InsertMaybe(atf_netio::FEndpoint& row) {
 652    ind_endpoint_Reserve(1);
 653    bool retval = true; // if already in hash, InsertMaybe returns true
 654    if (LIKELY(row.ind_endpoint_next == (atf_netio::FEndpoint*)-1)) {// check if in hash already
 655        u32 index = algo::Smallstr50_Hash(0, row.endpoint) & (_db.ind_endpoint_buckets_n - 1);
 656        atf_netio::FEndpoint* *prev = &_db.ind_endpoint_buckets_elems[index];
 657        do {
 658            atf_netio::FEndpoint* ret = *prev;
 659            if (!ret) { // exit condition 1: reached the end of the list
 660                break;
 661            }
 662            if ((*ret).endpoint == row.endpoint) { // exit condition 2: found matching key
 663                retval = false;
 664                break;
 665            }
 666            prev = &ret->ind_endpoint_next;
 667        } while (true);
 668        if (retval) {
 669            row.ind_endpoint_next = *prev;
 670            _db.ind_endpoint_n++;
 671            *prev = &row;
 672        }
 673    }
 674    return retval;
 675}

 677// --- atf_netio.FDb.ind_endpoint.Remove
 678// Remove reference to element from hash index. If element is not in hash, do nothing
 679void atf_netio::ind_endpoint_Remove(atf_netio::FEndpoint& row) {
 680    if (LIKELY(row.ind_endpoint_next != (atf_netio::FEndpoint*)-1)) {// check if in hash already
 681        u32 index = algo::Smallstr50_Hash(0, row.endpoint) & (_db.ind_endpoint_buckets_n - 1);
 682        atf_netio::FEndpoint* *prev = &_db.ind_endpoint_buckets_elems[index]; // addr of pointer to current element
 683        while (atf_netio::FEndpoint *next = *prev) {                          // scan the collision chain for our element
 684            if (next == &row) {        // found it?
 685                *prev = next->ind_endpoint_next; // unlink (singly linked list)
 686                _db.ind_endpoint_n--;
 687                row.ind_endpoint_next = (atf_netio::FEndpoint*)-1;// not-in-hash
 688                break;
 689            }
 690            prev = &next->ind_endpoint_next;
 691        }
 692    }
 693}

 695// --- atf_netio.FDb.ind_endpoint.Reserve
 696// Reserve enough room in the hash for N more elements. Return success code.
 697void atf_netio::ind_endpoint_Reserve(int n) {
 698    u32 old_nbuckets = _db.ind_endpoint_buckets_n;
 699    u32 new_nelems   = _db.ind_endpoint_n + n;
 700    // # of elements has to be roughly equal to the number of buckets
 701    if (new_nelems > old_nbuckets) {
 702        int new_nbuckets = i32_Max(algo::BumpToPow2(new_nelems), u32(4));
 703        u32 old_size = old_nbuckets * sizeof(atf_netio::FEndpoint*);
 704        u32 new_size = new_nbuckets * sizeof(atf_netio::FEndpoint*);
 705        // allocate new array. we don't use Realloc since copying is not needed and factor of 2 probably
 706        // means new memory will have to be allocated anyway
 707        atf_netio::FEndpoint* *new_buckets = (atf_netio::FEndpoint**)algo_lib::malloc_AllocMem(new_size);
 708        if (UNLIKELY(!new_buckets)) {
 709            FatalErrorExit("atf_netio.out_of_memory  field:atf_netio.FDb.ind_endpoint");
 710        }
 711        memset(new_buckets, 0, new_size); // clear pointers
 712        // rehash all entries
 713        for (int i = 0; i < _db.ind_endpoint_buckets_n; i++) {
 714            atf_netio::FEndpoint* elem = _db.ind_endpoint_buckets_elems[i];
 715            while (elem) {
 716                atf_netio::FEndpoint &row        = *elem;
 717                atf_netio::FEndpoint* next       = row.ind_endpoint_next;
 718                u32 index          = algo::Smallstr50_Hash(0, row.endpoint) & (new_nbuckets-1);
 719                row.ind_endpoint_next     = new_buckets[index];
 720                new_buckets[index] = &row;
 721                elem               = next;
 722            }
 723        }
 724        // free old array
 725        algo_lib::malloc_FreeMem(_db.ind_endpoint_buckets_elems, old_size);
 726        _db.ind_endpoint_buckets_elems = new_buckets;
 727        _db.ind_endpoint_buckets_n = new_nbuckets;
 728    }
 729}
