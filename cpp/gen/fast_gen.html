 775// --- fast.FDb.ind_type_name.InsertMaybe
 776// Insert row into hash table. Return true if row is reachable through the hash after the function completes.
 777bool fast::ind_type_name_InsertMaybe(fast::FType& row) {
 778    ind_type_name_Reserve(1);
 779    bool retval = true; // if already in hash, InsertMaybe returns true
 780    if (LIKELY(row.ind_type_name_next == (fast::FType*)-1)) {// check if in hash already
 781        u32 index = algo::Smallstr50_Hash(0, row.name) & (_db.ind_type_name_buckets_n - 1);
 782        fast::FType* *prev = &_db.ind_type_name_buckets_elems[index];
 783        do {
 784            fast::FType* ret = *prev;
 785            if (!ret) { // exit condition 1: reached the end of the list
 786                break;
 787            }
 788            if ((*ret).name == row.name) { // exit condition 2: found matching key
 789                retval = false;
 790                break;
 791            }
 792            prev = &ret->ind_type_name_next;
 793        } while (true);
 794        if (retval) {
 795            row.ind_type_name_next = *prev;
 796            _db.ind_type_name_n++;
 797            *prev = &row;
 798        }
 799    }
 800    return retval;
 801}

 803// --- fast.FDb.ind_type_name.Remove
 804// Remove reference to element from hash index. If element is not in hash, do nothing
 805void fast::ind_type_name_Remove(fast::FType& row) {
 806    if (LIKELY(row.ind_type_name_next != (fast::FType*)-1)) {// check if in hash already
 807        u32 index = algo::Smallstr50_Hash(0, row.name) & (_db.ind_type_name_buckets_n - 1);
 808        fast::FType* *prev = &_db.ind_type_name_buckets_elems[index]; // addr of pointer to current element
 809        while (fast::FType *next = *prev) {                          // scan the collision chain for our element
 810            if (next == &row) {        // found it?
 811                *prev = next->ind_type_name_next; // unlink (singly linked list)
 812                _db.ind_type_name_n--;
 813                row.ind_type_name_next = (fast::FType*)-1;// not-in-hash
 814                break;
 815            }
 816            prev = &next->ind_type_name_next;
 817        }
 818    }
 819}

 821// --- fast.FDb.ind_type_name.Reserve
 822// Reserve enough room in the hash for N more elements. Return success code.
 823void fast::ind_type_name_Reserve(int n) {
 824    u32 old_nbuckets = _db.ind_type_name_buckets_n;
 825    u32 new_nelems   = _db.ind_type_name_n + n;
 826    // # of elements has to be roughly equal to the number of buckets
 827    if (new_nelems > old_nbuckets) {
 828        int new_nbuckets = i32_Max(algo::BumpToPow2(new_nelems), u32(4));
 829        u32 old_size = old_nbuckets * sizeof(fast::FType*);
 830        u32 new_size = new_nbuckets * sizeof(fast::FType*);
 831        // allocate new array. we don't use Realloc since copying is not needed and factor of 2 probably
 832        // means new memory will have to be allocated anyway
 833        fast::FType* *new_buckets = (fast::FType**)algo_lib::malloc_AllocMem(new_size);
 834        if (UNLIKELY(!new_buckets)) {
 835            FatalErrorExit("fast.out_of_memory  field:fast.FDb.ind_type_name");
 836        }
 837        memset(new_buckets, 0, new_size); // clear pointers
 838        // rehash all entries
 839        for (int i = 0; i < _db.ind_type_name_buckets_n; i++) {
 840            fast::FType* elem = _db.ind_type_name_buckets_elems[i];
 841            while (elem) {
 842                fast::FType &row        = *elem;
 843                fast::FType* next       = row.ind_type_name_next;
 844                u32 index          = algo::Smallstr50_Hash(0, row.name) & (new_nbuckets-1);
 845                row.ind_type_name_next     = new_buckets[index];
 846                new_buckets[index] = &row;
 847                elem               = next;
 848            }
 849        }
 850        // free old array
 851        algo_lib::malloc_FreeMem(_db.ind_type_name_buckets_elems, old_size);
 852        _db.ind_type_name_buckets_elems = new_buckets;
 853        _db.ind_type_name_buckets_n = new_nbuckets;
 854    }
 855}

2683// --- fast.Groupname.c_field.Remove
2684// Find element using linear scan. If element is in array, remove, otherwise do nothing
2685void fast::c_field_Remove(fast::Groupname& groupname, fast::FField& row) {
2686    if (bool_Update(row.groupname_c_field_in_ary,false)) {
2687        int lim = groupname.c_field_n;
2688        fast::FField* *elems = groupname.c_field_elems;
2689        // search backward, so that most recently added element is found first.
2690        // if found, shift array.
2691        for (int i = lim-1; i>=0; i--) {
2692            fast::FField* elem = elems[i]; // fetch element
2693            if (elem == &row) {
2694                int j = i + 1;
2695                size_t nbytes = sizeof(fast::FField*) * (lim - j);
2696                memmove(elems + i, elems + j, nbytes);
2697                groupname.c_field_n = lim - 1;
2698                break;
2699            }
2700        }
2701    }
2702}
