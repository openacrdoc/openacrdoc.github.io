 208// Parse a URL from string STR to OUT.
 209// The format of a URL is
 210// protocol://user:password@somehost:port<-source_interface/dir1/dir2
 211// URL fields are
 212// protocol, username, password, server, port, host, source_addr_host, dir
 213//
 214//
 215// Windows pathnames are supported, e.g.
 216// when parsing
 217//    file://c:/dir/dir2
 218// c: will not be parsed as a username, but as part of the pathname.
 219//
 220bool algo::URL_ReadStrptrMaybe(URL &out, algo::strptr str) {
 221    bool retval = true;
 222    Refurbish(out);
 223    out.port = -1;
 224    frep_(i,elems_N(str)-2) {
 225        if (str[i]==':' && str[i+1]=='/' && str[i+2]=='/') {    // find ://
 226            out.protocol = FirstN(str,i);
 227            str = RestFrom(str,i+3);
 228        }
 229        if (!algo_lib::IdentCharQ(str[i])) {
 230            break;
 231        }
 232    }
 233    if (out.port < 0) {
 234        if (StrEqual(out.protocol,"ftp", false)){
 235            out.port = 21;
 236        } else if (StrEqual(out.protocol,"http", false)){
 237            out.port = 80;
 238        }
 239    }
 240    if (StartsWithQ(out.protocol,"file") && elems_N(str)>1 && str[1] == ':') {  // file://c:/dir/dir2    -> recognizes c:
 241        out.dir = str;
 242    } else {
 243        i32_Range R = TFind(str,'/');
 244        if(StrEqual(out.protocol,"scp",false)){
 245            R = TFind(str,':');
 246        }
 247        out.dir = R.end > R.beg ? RestFrom(str,R.end - (R.beg>0)) : strptr(); // proto:///dir/foo  -- dir = dir/foo
 248        str     = FirstN(str,R.beg); // www.example.com:80
 249        R            = TRevFind(str, '@');
 250        strptr login = FirstN(str, R.beg);
 251        str          = RestFrom(str, R.end);
 252        R            = TFind(login,':');
 253        out.username = FirstN(login, R.beg);
 254        out.password = RestFrom(login, R.end);
 255        R          = substr_FindFirst(str, strptr("<-"));
 256        if (R.beg == R.end) R = substr_FindFirst(str, strptr("~-"));
 257        out.host   = RestFrom(str,R.end);
 258        str        = FirstN(str,R.beg);
 259        if (elems_N(str)>2 && str[0]=='(' && str[elems_N(str)-1]==')') {
 260            str=GetRegion(str,1,elems_N(str)-2);
 261            Ipmask mask;
 262            i32_Range comma = substr_FindLast(str, strptr(","));
 263            retval = retval && Ipmask_ReadStrptrMaybe(mask,FirstN(str, comma.beg));
 264            if (mask.ip_host != 0) {
 265                out.source_addr_host = mask.ip_host;
 266            }
 267            str = RestFrom(str, comma.end);
 268        }
 269        R      = TFind(str, ':');
 270        out.server = FirstN(str,R.beg);
 271        algo::StringIter iter(str,R.end);
 272        int ret = TryParseI32(iter, out.port);
 273        (void)ret; // coverity
 274    }
 275    return retval;
 276}

 319bool algo::Ipmask_ReadStrptrMaybe(Ipmask &row, algo::strptr str) {
 320    bool retval = true;
 321    algo::StringIter iter(str);
 322    int ip = 0;
 323    int mask = 0;
 324    int asterisks = 0;
 325    // this code parses expressions like 192.168.*.*
 326    // and 2130706433 (which is 127.0.0.1)
 327    // but it can't parse a combination of numbers that don't fit one byte and asterisks, such as
 328    // 1234.*.1234
 329    rep_(i,4) {
 330        if (SkipChar(iter, '*')) {
 331            asterisks++;
 332            ip=ip<<8;
 333            mask=mask<<8;
 334        } else {
 335            int piece;
 336            if (TryParseI32(iter, piece)) {
 337                ip=ip<<8|piece;
 338                mask=mask<<8|0xff;
 339            } else {
 340                return true;// do nothing
 341            }
 342        }
 343        if (i!=3 && !SkipChar(iter, '.')) {
 344            break;
 345        }
 346    }
 347    if (asterisks == 0) {
 348        mask = 0xffffffff;
 349    }
 350    if (SkipChar(iter, '/')) {
 351        u32 bits=0;
 352        int i=0;
 353        for(; i<4; i++) {
 354            int piece;
 355            if (TryParseI32(iter, piece)) {
 356                bits=bits<<8|piece;
 357            } else {
 358                return true;// do nothing
 359            }
 360            if (i!=3 && !SkipChar(iter, '.')) {
 361                break;
 362            }
 363        }
 364        if (i==4) {
 365        } else if (i==0) {
 366            if (bits!=0) {// special case: A.B.C.D/0 -> bits 0
 367                bits = u32(-1) << (32-bits);
 368            }
 369        }
 370        mask &= bits;
 371    }
 372    row.ip_host=ip;
 373    row.mask = mask;
 374    return retval;
 375}

 383// pads with zeros on the left so that at least 'atleast' characters are output.
 384void algo::i64_PrintPadLeft(i64 num, algo::cstring &out, int atleast) {
 385    i64 number=num;
 386    char c[32];
 387    int ndig = 0;
 388    int neg=number<0;
 389    number=Abs(number);
 390    if (!number) {
 391        c[ndig++]='0';
 392    } else {
 393        while (number!=0) {
 394            c[ndig++]=char('0' + number%10);
 395            number/=10;
 396        }
 397    }
 398    strptr result(ch_AllocN(out, i32_Max(ndig,atleast)+neg));
 399    int i = 0;
 400    if (neg) {  // minus sign
 401        result[i++] = '-';
 402        atleast--;
 403    }
 404    for (int k=ndig; k<atleast; k++) {  // pad with zeros from left
 405        result[i++]='0';
 406    }
 407    while (--ndig >= 0) {   // print the rest
 408        result[i++]=c[ndig];
 409    }
 410}

 414// Assuming SRC is a number, Transfer SRC to OUT, inserting
 415// commas between groups of 3 numbers.
 416// 0 -> 0
 417// 11 -> 11
 418// 222 -> 222
 419// 3333 -> 3,333
 420// 4567.3 -> 4,567.3
 421// 1.11111 -> 1.11111
 422void algo::strptr_PrintWithCommas(strptr src, algo::cstring &out) {
 423    int dec = Find(src, '.');
 424    if (dec==-1) {
 425        dec=src.n_elems;
 426    }
 427    ch_Reserve(out, src.n_elems * 2);
 428    int i=0;
 429    if (src.elems[0]=='-') {
 430        out.ch_elems[out.ch_n++] = src.elems[i++];
 431    }
 432    int remaining=(dec-i)%3;
 433    if (remaining==0) {
 434        remaining += 3;
 435    }
 436    for (; i<dec; i++) {
 437        if (remaining==0) {
 438            out.ch_elems[out.ch_n++] = ',';
 439            remaining += 3;
 440        }
 441        remaining--;
 442        out.ch_elems[out.ch_n++] = src.elems[i];
 443    }
 444    // print remainder of the string as-is
 445    out << RestFrom(src,dec);
 446}

 448// -----------------------------------------------------------------------------

 450// Assuming STR is a number, remove any unnecessary characters from the right of it.
 451// Unnecessary characters are trailing zeros after a dot.
 452// If the string contains no dot, nothing is done.
 453// If a trailing '.' or a single '-' remains, it is removed as well.
 454// If the resulting string is empty, a single zero is returned.
 455//
 456// 0         -> 0
 457// 0.1       -> 0.1
 458// 0.0       -> 0
 459// 12345.000 -> 12345
 460// -0        -> -0
 461// .0        -> 0
 462// -.0       -> 0
 463// -0.0      -> -0
 464// -10.0     -> -10
 465//           ->            empty string is not touched
 466// Since the string may be edited, you can't pass compile-time constants
 467// to this function. In fact the only reason this function is not private to algo_lib
 468// is because of unit testing.
 469// If the initial string is empty, nothing is done.
 470// BEWARE: this function will happily convert 1e+60 to 1e+6
 471//
 472void algo::strptr_TrimZerosRight(strptr &str) {
 473    int n=str.n_elems;
 474    int dotpos=0;
 475    for (; dotpos < str.n_elems; dotpos++) {
 476        if (str.elems[dotpos] == '.') {
 477            break;
 478        }
 479    }
 480    if (n > 0) {
 481        // kill trailing zeros, but don't leave an empty string
 482        while (n > dotpos && str.elems[n-1] == '0') {
 483            n--;
 484        }
 485        // kill trailing .
 486        if (n > 0 && str.elems[n-1] == '.') {
 487            n--;
 488        }
 489        // kill trailing -
 490        // (this effectively kills -. as well)
 491        if (n > 0 && str.elems[n-1] == '-') {
 492            n--;
 493        }
 494        // fix empty string
 495        if (n == 0) {
 496            str.elems[n++] = '0';
 497        }
 498        str.n_elems=n;
 499    }
 500}

 519// Print double D into string OUT with PRECISION digits
 520// after the decimal point.
 521// If OMIT_ZEROS is specified, trailing zeros that are safe to omit are omitted.
 522// If COMMAS is specified, the large numbers are printed in groups of 3 digits
 523// with commas between them.
 524void algo::double_PrintPrec(double d, algo::cstring &out, int precision, bool omit_zeros, bool commas) {
 525    char buf[512];
 526    precision = Clipped(precision,20);
 527    // create a format string for the specified precision
 528    // i.e. %.7lf
 529    char fmt[8];
 530    int i=0;
 531    fmt[i++] = '%';
 532    fmt[i++] = '.';
 533    if (precision < 10) {
 534        fmt[i++] = '0'+precision;
 535    } else {
 536        fmt[i++] = '1';
 537        fmt[i++] = '0'+precision-10;
 538    }
 539    fmt[i++] = 'l';
 540    fmt[i++] = 'f';
 541    fmt[i++] = 0;
 542    int n=snprintf(buf,sizeof(buf),fmt,d);
 543    strptr result(buf,n);
 544    if (ScientificQ(result)) {
 545        omit_zeros=false;
 546        commas=false;
 547    }
 548    if (omit_zeros) {
 549        strptr_TrimZerosRight(result);
 550    }
 551    if (commas) {
 552        strptr_PrintWithCommas(result,out);
 553    } else {
 554        out << result;
 555    }
 556}

 918
 919// Print scaled decimal.
 920// VALUE           integer value
 921// SCALE           log10 to get the real value: real value = value / pow(10,scale)
 922// BUF             target buffer. buffer must be large enough.
 923// FIXED           if true, make sure there are exactly SCALE characters after decimal point (and always emit decimal point)
 924// RETURN          number of characters printed.
 925// Example:
 926// u64_FmtBufDec(0    , 2, .. false ) -> "0"
 927// u64_FmtBufDec(0    , 2, .. true  ) -> "0.00"
 928// u64_FmtBufDec(123  , 1, .. false ) -> "12.3"
 929// u64_FmtBufDec(123  , 2, .. false ) -> "1.23"
 930// u64_FmtBufDec(123  , 2, .. true  ) -> "1.23"
 931// u64_FmtBufDec(1230 , 3, .. false ) -> "1.23"
 932// u64_FmtBufDec(1230 , 3, .. true  ) -> "1.230"
 933u32 algo::u64_FmtBufDec(u64 num, int scale, u8 *buf, bool fixed) {
 934    char temp_elems[32];// output in reverse order
 935    int temp_n        = 0;
 936    int skip_zeros    = !fixed;//
 937    int first_nonzero = 0;
 938    do {
 939        int k = int(num%10);//the digit
 940        num     = num/10;
 941        if (k==0) {
 942            first_nonzero += skip_zeros;// index of character where
 943        }
 944        skip_zeros &= k==0; // count # of leading zeros
 945        temp_elems[temp_n++]=char('0'+k);
 946        if (temp_n==scale) {
 947            temp_elems[temp_n++] = '.';
 948            skip_zeros=false;
 949        }
 950    } while (num > 0);
 951    if (temp_n < scale) {
 952        while (temp_n<scale) {
 953            temp_elems[temp_n++]='0';
 954            first_nonzero += skip_zeros;
 955        }
 956        temp_elems[temp_n++] = '.';
 957        temp_elems[temp_n++] = '0';
 958    } else if (temp_n==scale+1) {
 959        temp_elems[temp_n++] = '0';
 960    }
 961    int num_chars = temp_n - first_nonzero;
 962    u8 *start = buf;
 963    while (temp_n > first_nonzero) {// transfer over & skip trailing zeros
 964        *start++ = temp_elems[--temp_n];
 965    }
 966    num_chars -= buf[num_chars-1]=='.';// remove trailing .
 967    return num_chars;
 968}

 990// print TIME to STR, according to spec SPEC
 991// The following control characters are supported in SPEC.
 992// %Y     Year printed as 4 digits, e.g. 2002
 993// %y     Last 2 digits of year printed as 2 digits
 994// %B     Long month name (February)
 995// %b     Short month name (Feb)
 996// %a     Week day name (Monday)
 997// %m     Month printed as 2 digits, with January being 01
 998// %d     Day of month printed as at least 2 digits (or WIDTH)
 999// %.X    where X is a number -- set width
1000// %I     Hour using 12-hour format
1001// %H     Hour using 24-hour format, printed as 2 digits
1002// %M     Minute, printed as 2 digits
1003// %S     Second, printed as 2 digits
1004// %T     Shortcut for %H:%M:%S.%.9X
1005// %X     Print fractions of a second according to width
1006// %x     Print milliseconds with 3 decimal places
1007// %-     Print - sign if timespec is negative
1008// %%     Print % sign
1009void algo::TimeStruct_Print(const TimeStruct &time, algo::cstring &str, const algo::strptr &spec) {
1010    for(int i =0; i<elems_N(spec);) {
1011        int selector = spec[i++];
1012        if (selector != '%') {
1013            str<<char(selector);
1014            continue;
1015        }
1016        int width = 0;
1017        if (i+1 < elems_N(spec)-1 && spec[i]=='.' && algo_lib::DigitCharQ(spec[i+1])) {
1018            width = spec[i+1] - '0';
1019            i+=2;
1020        }
1021        if (i == elems_N(spec)) break;
1022        selector = spec[i++];
1023        switch(selector) {
1024        case 'Y':
1025            i64_PrintPadLeft(time.tm_year+1900, str, 4);
1026            break;
1027        case 'y':
1028            i64_PrintPadLeft(time.tm_year+1900>=2000 ? time.tm_year-100 : time.tm_year, str, 2);
1029            break;
1030        case 'B':
1031            str<<GetMonthNameZeroBased(Clipped(time.tm_mon,12));
1032            break;
1033        case 'b':
1034            str<<GetRegion(GetMonthNameZeroBasedShort(Clipped(time.tm_mon,12)), 0,3);
1035            break;
1036        case 'a':
1037            str<<GetWeekdayName(Clipped(time.tm_wday,7));
1038            break;
1039        case 'm':
1040            i64_PrintPadLeft(int(time.tm_mon)+1, str, i32_Max(width,2));
1041            break;
1042        case 'd':
1043            i64_PrintPadLeft(int(time.tm_mday), str, i32_Max(width,2));
1044            break;
1045        case 'I':
1046            i64_PrintPadLeft(time.tm_hour > 12 ? time.tm_hour - 12 : time.tm_hour, str, 2);
1047            break;
1048        case 'H':
1049            i64_PrintPadLeft(time.tm_hour, str, 2);
1050            break;
1051        case 'M':
1052            i64_PrintPadLeft(time.tm_min, str, 2);
1053            break;
1054        case 'S':
1055            i64_PrintPadLeft(time.tm_sec, str, 2);
1056            break;
1057        case 'T':
1058            i64_PrintPadLeft(time.tm_hour, str, 2);
1059            str<<":";
1060            i64_PrintPadLeft(time.tm_min, str, 2);
1061            str<<":";
1062            i64_PrintPadLeft(time.tm_sec, str, 2);
1063            if (time.tm_nsec) {
1064                str << '.';
1065                i32_PrintNanosec(time.tm_nsec, str, width, 9);
1066            }
1067            break;
1068        case 'X':
1069            {
1070                int div = 1;
1071                switch(9-width) {
1072                case 0: div= 1; break;
1073                case 1: div= 10; break;
1074                case 2: div= 100; break;
1075                case 3: div= 1000; break;
1076                case 4: div= 10000; break;
1077                case 5: div= 100000; break;
1078                case 6: div= 1000000; break;
1079                case 7: div= 10000000; break;
1080                case 8: div= 100000000; break;
1081                case 9: div= 1; break;// default?????
1082                default: div= 1; break;
1083                }
1084                i32_PrintNanosec(time.tm_nsec / div, str, width, 9);
1085            }
1086            break;
1087        case 'x':
1088            i32_PrintNanosec(time.tm_nsec / 1000000, str, width, 3);
1089            break;
1090        case '-':
1091            if (time.tm_neg) str << '-';
1092            break;
1093        case '%':
1094            str<<'%';
1095            break;
1096        default:
1097            str<<char(selector);
1098            break;
1099        }
1100    }

1316static void Normalize(algo_lib::FTxttbl &txttbl) {
1317    algo::U64Ary norm;
1318    ary_Alloc(norm) = 0;
1319    int maxspan = 0;
1320    ind_beg(algo_lib::txttbl_c_txtrow_curs,txtrow,txttbl) if (txtrow.select) {
1321        ind_beg(algo_lib::txtrow_c_txtcell_curs, txtcell, txtrow) {
1322            maxspan = i32_Max(maxspan, txtcell.span);
1323        }ind_end;
1324    }ind_end;
1325    frep_(span,maxspan+1) {
1326        ind_beg(algo_lib::txttbl_c_txtrow_curs,txtrow,txttbl) if (txtrow.select) {
1327            int i = 0;
1328            ind_beg(algo_lib::txtrow_c_txtcell_curs, txtcell, txtrow) {
1329                if (txtcell.span == span) {
1330                    int w = i32_Max(ch_N(txtcell.text) + ch_N(txtcell.rsep), txtcell.width);
1331                    // apply width
1332                    int r = i + txtcell.span;
1333                    while (r >= ary_N(norm)) {
1334                        int last = ary_qLast(norm);
1335                        ary_Alloc(norm) = last;
1336                    }
1337                    int extra = w - (ary_qFind(norm, r) - ary_qFind(norm, i));
1338                    if (extra > 0) {
1339                        for (int j = r; j < ary_N(norm); j++) {
1340                            ary_qFind(norm, j) += extra;
1341                        }
1342                    }
1343                }
1344                i += txtcell.span;
1345            }ind_end;
1346        }ind_end;
1347    }
1348    ind_beg(algo_lib::txttbl_c_txtrow_curs,txtrow,txttbl) if (txtrow.select) {
1349        int i = 0;
1350        ind_beg(algo_lib::txtrow_c_txtcell_curs, txtcell, txtrow) {
1351            txtcell.width = ary_qFind(norm, i+txtcell.span) - ary_qFind(norm, i);
1352            i += txtcell.span;
1353        }ind_end;;
1354    }ind_end;;
1355}

1607// basis function for reading tuples.

1621                break;

1788// Write character C into buffer BUF, using C++ character escapement rules
1789// BUF must have room for at least 4 extra characters
1790// Return number of characters written
1791// QUOTE_CHAR also gets escaped.
1792static inline int _PrintQuotedChar(char c, char *buf, char quote_char) {
1793    int n=0;
1794    if (UNLIKELY((c==quote_char) | (c=='\\'))) {
1795        buf[0]='\\';
1796        buf[1]=c;
1797        n=2;
1798    } else if (UNLIKELY((unsigned(c) < 32) | (unsigned(c) >= 127))) {
1799        if (c=='\n') {// 10
1800            buf[0]='\\';
1801            buf[1]='n';
1802            n=2;
1803        } else if (c=='\t') {// 9
1804            buf[0]='\\';
1805            buf[1]='t';
1806            n=2;
1807        } else if (c=='\r') {// 13
1808            buf[0]='\\';
1809            buf[1]='r';
1810            n=2;
1811        } else {
1812            unsigned x = c&0xff;
1813            buf[0]='\\';
1814            buf[1]='0' + ((x>>6) & 7);
1815            buf[2]='0' + ((x>>3) & 7);
1816            buf[3]='0' + ((x>>0) & 7);
1817            n=4;
1818        }
1819    } else {
1820        buf[0]=c;
1821        n=1;
1822    }
1823    return n;
1824}

1840void algo::strptr_PrintCppQuoted(algo::strptr str, algo::cstring &out, char quote_char) {
1841    ch_Reserve(out, 2+str.n_elems*4);// max possible
1842    char *srcbuf=str.elems;// source string
1843    char *buf = out.ch_elems;// destination buffer
1844    int src_n=str.n_elems;// source count of chars
1845    int n=out.ch_n;// destination count of chars
1846    buf[n++] = quote_char;// opening quote
1847    for (int i=0; i<src_n; i++) {
1848        n += _PrintQuotedChar(srcbuf[i],buf+n,quote_char);
1849    }
1850    buf[n++] = quote_char;// closing quote
1851    out.ch_n = n;
1852}

1933// Convert VALUE to a string, encoded as a u64.
1934// Value must be >=0 and <100000000.
1935// Resulting u64 is little-endian -- MSB contains first byte
1936// of the resulting string. To be used as an ascii string, value
1937// 0x3030303030303030 (8 ascii '0's) must be added to the result.
1938// Don't use this function. Use u32_FmtBufSmall, and u64_FmtBuf.
1939u64 algo::u32_CvtLestr8Small(u32 value) {
1940    u32 l = value % 10000;
1941    u32 h = value / 10000;
1942
1943    u32 ll = l % 100;
1944    u32 lh = l / 100;
1945    u32 hl = h % 100;
1946    u32 hh = h / 100;
1947
1948    u32 lll = ll % 10;
1949    u32 llh = ll / 10;
1950    u32 lhl = lh % 10;
1951    u32 lhh = lh / 10;
1952    u32 hll = hl % 10;
1953    u32 hlh = hl / 10;
1954    u32 hhl = hh % 10;
1955    u32 hhh = hh / 10;
1956
1957    return
1958        ( (u64)lll << 56
1959          | (u64)llh << 48
1960          | (u64)lhl << 40
1961          | (u64)lhh << 32
1962          | (u64)hll << 24
1963          | (u64)hlh << 16
1964          | (u64)hhl << 8
1965          | (u64)hhh << 0);
1966}

1969
1970// Format VALUE, which should be >=0 and <100000000,
1971// as a string and write it to buffer BUF. Return length of resulting string.
1972// Buffer must be big enough.
1973u32 algo::u32_FmtBufSmall(u32 value, u8 *buf) {
1974    u64 result = u32_CvtLestr8Small(value);
1975    u64 zeros  = u64_BitScanForward(result);
1976    zeros      = value == 0 ? 56 : zeros;
1977    u64 len    = 8 - zeros / 8;
1978    *(u64*)buf = (result + 0x3030303030303030) >> zeros/8*8;
1979    return u32(len);
1980}

1983
1984// Format VALUE, which is an arbitrary u64,
1985// as a string and write it to buffer BUF.
1986// Return length of resulting string.
1987// Buffer must be big enough.
1988u32 algo::u64_FmtBuf(u64 value, u8 *buf) {
1989    static const u32 thresh = 100000000;
1990    u32 len=0;
1991    if (value < thresh) {
1992        return u32_FmtBufSmall((u32)value, buf);
1993    }
1994    u64 r1 = value % thresh;
1995    value  = value / thresh;
1996    if (value < thresh) {
1997        len              = u32_FmtBufSmall((u32)value,buf);
1998        *(u64*)(buf+len) = u32_CvtLestr8Small((u32)r1) + 0x3030303030303030;
1999        return len+8;
2000    }
2001    u64 r2             = value % thresh;
2002    u64 r3             = value / thresh;
2003    len                = u32_FmtBufSmall((u32)r3,buf);
2004    *(u64*)(buf+len)   = u32_CvtLestr8Small((u32)r2) + 0x3030303030303030;
2005    *(u64*)(buf+len+8) = u32_CvtLestr8Small((u32)r1) + 0x3030303030303030;
2006    return len+8+8;
2007}

2013// Return length of resulting string.
2014// Buffer must be big enough.
2015u32 algo::u32_FmtBuf(u32 value, u8 *buf) {
