 27static void JsonParseError(lib_json::FParser &parser, strptr info) {
 28    if (lib_json::JsonParseOkQ(parser)) {
 29        parser.state      = lib_json_FParser_state_err;
 30        parser.err_info   = info;
 31        parser.err_offset = parser.offset + parser.ind;
 32    }
 33}

 35// Note: this detects duplicate field names.
 36inline bool NodeParentingRulesOkQ(lib_json::FNode *parent, lib_json_FNode_type_Enum type, strptr value) {
 37    bool ok = false;
 38    bool isfield = type == lib_json_FNode_type_field;
 39    ok     |= parent == NULL && type != lib_json_FNode_type_field;
 40    ok     |= (parent && parent->type == lib_json_FNode_type_array  && !isfield);
 41    ok     |= (parent && parent->type == lib_json_FNode_type_field  && !isfield && !c_child_N(*parent));
 42    ok     |= (parent && parent->type == lib_json_FNode_type_object && isfield
 43               && !ind_objfld_Find(lib_json::FldKey(parent->p_parent,value)));
 44    return ok;
 45}

110// add new node as child of current node (or as root node)
111// NULL on error
112static lib_json::FNode *JsonNodeAdd(lib_json::FParser &parser, lib_json_FNode_type_Enum type, strptr value=strptr()) {
113    if (!NodeParentingRulesOkQ(parser.node,type,value)) {
114        JsonParseError(parser,"misplaced colon");
115        return NULL;
116    }
117    lib_json::FNode &node = qNewNode(parser.node,type,value);
118    if (!parser.root_node) {
119        parser.root_node = &node;
120        parser.node      = &node;
121    }
122    return &node;
123}

125// move one level up in tree
126static void JsonUp(lib_json::FParser &parser) {
127    if (parser.node) {
128        parser.node = (parser.node->p_parent != parser.node ? parser.node->p_parent : NULL);
129    } else {
130        JsonParseError(parser,"closing curly brace or bracket has no opening match");
131    }
132}

134// Open object
135static void JsonObjectOpen(lib_json::FParser &parser) {
136    parser.node = JsonNodeAdd(parser,lib_json_FNode_type_object);
137    parser.need_comma = false;
138    parser.have_comma = false;
139}

141// Close object
142static void JsonObjectClose(lib_json::FParser &parser) {
143    if (!parser.need_comma && parser.have_comma) {
144        JsonParseError(parser,"value expected");
145        return;
146    }
147    if (parser.node && parser.node->type == lib_json_FNode_type_field) {
148        if (!c_child_N(*parser.node)) {
149            JsonParseError(parser,"field value expected");
150            return;
151        }
152        JsonUp(parser);
153    }
154    if (parser.node && parser.node->type == lib_json_FNode_type_object) {
155        parser.need_comma = true;
156        parser.have_comma = false;
157        JsonUp(parser);
158    } else {
159        JsonParseError(parser,"closing curly brace mismatch");
160    }
161}

185// New field
186static void JsonField(lib_json::FParser &parser) {
187    if (!parser.need_colon) {
188        JsonParseError(parser,"unexpected colon");
189    }
190    parser.need_colon = false;
191    if (!parser.node || !(parser.node->type == lib_json_FNode_type_field)) {
192        JsonParseError(parser,"unexpected colon");
193    }

196// Next element
197static void JsonNextElem(lib_json::FParser &parser) {
198    if (!parser.need_comma) {
199        JsonParseError(parser,"unexpected comma");
200        return;
201    }
202    parser.need_comma = false;
203    parser.have_comma = true;
204    if (parser.node && parser.node->type == lib_json_FNode_type_field) {
205        JsonUp(parser);
206    }
207    bool ok = parser.node
208        && (parser.node->type == lib_json_FNode_type_object
209            || parser.node->type == lib_json_FNode_type_array);
210    if (!ok) {
211        JsonParseError(parser,"unexpected comma");
212    }
213}

222// Handle string
223static void JsonString(lib_json::FParser &parser) {
224    if (parser.node && parser.node->type == lib_json_FNode_type_object) {
225        parser.node = JsonNodeAdd(parser,lib_json_FNode_type_field,parser.value);
226        parser.need_colon = true;
227    } else {
228        JsonNodeAdd(parser,lib_json_FNode_type_string,parser.value);
229        parser.need_comma = true;
230        parser.have_comma = false;
231    }

234// Handle token
235static void JsonToken(lib_json::FParser &parser) {
236    if (parser.value == "true") {
237        JsonNodeAdd(parser,lib_json_FNode_type_true);
238        parser.need_comma = true;
239        parser.have_comma = false;
240    } else if (parser.value == "false") {
241        JsonNodeAdd(parser,lib_json_FNode_type_false);
242        parser.need_comma = true;
243        parser.have_comma = false;
244    } else if (parser.value == "null") {
245        JsonNodeAdd(parser,lib_json_FNode_type_null);
246        parser.need_comma = true;
247        parser.have_comma = false;
248    } else if (parser.value == "throw" && !parser.root_node) {
249        parser.state = lib_json_FParser_state_sec_line;
250    } else {
251        JsonParseError(parser,tempstr()<<"bad token: "<<parser.value);
252    }
253}

257    for (; parser.ind<elems_N(parser.buf); ++parser.ind) {
258        if (!algo_lib::WhiteCharQ(parser.buf[parser.ind])) {

265// Parse number
266static void JsonParseNumber(lib_json::FParser &parser) {
267    bool done = !elems_N(parser.buf);
268    i32  beg  = parser.ind;
269    for (;parser.ind<elems_N(parser.buf); ++parser.ind) {
270        if (!lib_json::JsonNumCharQ(parser.buf[parser.ind])) {
271            done = true;
272            break;
273        }
274    }
275    ch_Addary(parser.value,RestFrom(FirstN(parser.buf,parser.ind),beg));
276    if (done) {
277        parser.state = lib_json_FParser_state_ws;
278        JsonNumber(parser);
279        ch_RemoveAll(parser.value);
280    }
281}

283// encode as utf-8
284// todo check and not encode:
285// d800-dfff  : UTF-16 uses d800-dfff to encode it's surrogate pairs, i.e. values that don't fit in to 16 bits.
286//              This means UTF-8/UTF-32 are not allowed to encode these values.
287// fdd0-fdef  : Non-characters.
288// xfffe-xffff: Unassigned: x ranges from 0 to 10 (hex), so the values to check for are fffe-ffff, 1fffe-1ffff, etc.
289
290static void Utf8Encode(u32 c, cstring &out) {
291    if (c<0x80) {
292        out << char(c);
293    } else if (c<0x0800) {
294        out << char((c>>6&0x1f)|0xc0)  << char((c&0x3f)|0x80);
295    } else if (c<0x010000) {
296        out << char((c>>12&0x0f)|0xe0) << char((c>>6&0x3f)|0x80)  << char((c>>0&0x3f)|0x80);
297    } else if (c<0x110000) {
298        out << char((c>>18&0x07)|0xf0) << char((c>>12&0x3f)|0x80) << char((c>>6&0x3f)|0x80) << char((c>>0&0x3f)|0x80);
299    }
300    // 10ffff is the highest defined codepoint
301}

314// expect low surrogate u
315static void JsonParseStringSurrogate2(lib_json::FParser &parser) {
316    if (parser.ind<elems_N(parser.buf)) {
317        if (parser.buf[parser.ind++] == 'u') {
318            parser.state = lib_json_FParser_state_str_uesc;
319            parser.uesc_need = 4;

326// parse unicode string escape
327static void JsonParseStringUnicode(lib_json::FParser &parser) {
328    bool ok = true;
329    for (;ok && parser.uesc_need && parser.ind<elems_N(parser.buf);--parser.uesc_need,++parser.ind) {
330        u8 dig = 0;
331        ok &= algo::ParseHex1(parser.buf[parser.ind],dig) == 1;
332        parser.uesc_value <<= 4;
333        parser.uesc_value  |= dig;
334    }
335    if (!ok) {
336        tempstr info("unrecognized character within unicode char escape: code \\x");
337        u64_PrintHex(parser.buf[parser.ind-1],info,2,false,true);
338        JsonParseError(parser,info);
339        return;
340    }
341    if (!parser.uesc_need) {
342        // decode utf-16
343        if (parser.uesc_value<0xd800) {
344            // BMP chars
345            Utf8Encode(parser.uesc_value,parser.value);
346            parser.uesc_value = 0;
347            parser.state = lib_json_FParser_state_str;
348        } else if (parser.uesc_value<0xdc00) {
349            // high surrogate
350            parser.state = lib_json_FParser_state_str_usurr1;
351        } else if (parser.uesc_value<0xe000) {
352            // low surrogate, unexpected
353            tempstr info("UTF-16: unexpected low surrogate: \\u");
354            u64_PrintHex(parser.uesc_value,info,4,false,true);
355            JsonParseError(parser,info);
356        } else if (parser.uesc_value<0x10000) {
357            // rest BMP chars
358            Utf8Encode(parser.uesc_value,parser.value);
359            parser.uesc_value = 0;
360            parser.state = lib_json_FParser_state_str;
361        } else if ((parser.uesc_value&0xfc00) == 0xdc00) {
362            // high word - high surrogate, low word - low surrogate
363            u32 value = parser.uesc_value>>6&0x03fffc00;
364            value    += parser.uesc_value&0xffff;
365            value    -= 0x35FDC00;
366            Utf8Encode(value,parser.value);
367            parser.uesc_value = 0;
368            parser.state = lib_json_FParser_state_str;
369        } else {
370            // error, not a low surrogate
371            tempstr info("UTF-16: bad low surrogate: \\u");
372            u64_PrintHex(parser.uesc_value&0xffff,info,4,false,true);
373            JsonParseError(parser,info);
374        }
375    }
376}

378// parse string escape
379static void JsonParseStringEscape(lib_json::FParser &parser) {
380    if (parser.ind<elems_N(parser.buf)) {
381        parser.state = lib_json_FParser_state_str;
382        switch (parser.buf[parser.ind]) {
383        case '"' :
384        case '\\':
385        case '/' : parser.value << parser.buf[parser.ind]; break;
386        case 'b' : parser.value << '\b';     break;
387        case 'f' : parser.value << '\f';     break;
388        case 'n' : parser.value << '\n';     break;
389        case 'r' : parser.value << '\r';     break;
390        case 't' : parser.value << '\t';     break;
391        case 'u' : parser.state = lib_json_FParser_state_str_uesc; parser.uesc_need = 4; break;
392        default  : {
393            tempstr info;
394            info << "unrecognized string escape character: code \\x";
395            u64_PrintHex(parser.buf[parser.ind],info,2,false,true);
396            JsonParseError(parser,info);
397            return;
398        }
399        }
400        ++parser.ind;
401    }
402}

404// parse string
405static void JsonParseString(lib_json::FParser &parser) {
406    bool done = false;
407    for (; !done && parser.ind<elems_N(parser.buf); ++parser.ind) {
408        switch(parser.buf[parser.ind]) {
409        case '\\': parser.state = lib_json_FParser_state_str_esc; ++parser.ind; return; // return immediately
410        case '"' : done = true;                            break;
411        default  : parser.value << parser.buf[parser.ind]; break;
412        }
413    }
414    if (done) {
415        parser.state = lib_json_FParser_state_ws;
416        JsonString(parser);
417        ch_RemoveAll(parser.value);
418    }
419}

421// parse token
422static void JsonParseToken(lib_json::FParser &parser) {
423    bool done = !elems_N(parser.buf);
424    i32 beg = parser.ind;
425    for (;parser.ind<elems_N(parser.buf); ++parser.ind) {
426        if (!algo_lib::AlphaCharQ(parser.buf[parser.ind])) {
427            done = true;
428            break;
429        }

432    if (done) {
433        parser.state = lib_json_FParser_state_ws;
434        JsonToken(parser);
435        ch_RemoveAll(parser.value);
436    }

439// parse security line
440static void JsonParseSecLine(lib_json::FParser &parser) {
441    bool done = !elems_N(parser.buf);
442    for (;parser.ind<elems_N(parser.buf); ++parser.ind) {
443        if (parser.buf[parser.ind] == ';') {
444            ++parser.ind;

446            break;
447        }

450        parser.state = lib_json_FParser_state_ws;
451    }

454// parse value
455static void JsonParseValue(lib_json::FParser &parser) {
456    if (parser.ind<elems_N(parser.buf)) {
457        char c = parser.buf[parser.ind];
458        bool ok = !parser.need_comma || c==',' || c==']' || c=='}';
459        if (!ok) {
460            JsonParseError(parser,"comma expected");
461            return;
462        }
463        ok = !parser.need_colon || c==':';
464        if (!ok) {
465            JsonParseError(parser,"colon expected");
466            return;
467        }
468        switch(c) {
469        case '{':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonObjectOpen(parser);  break;
470        case '}':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonObjectClose(parser); break;
471        case '[':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonArrayOpen(parser);   break;
472        case ']':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonArrayClose(parser);  break;
473        case '"':   ++parser.ind; parser.state = lib_json_FParser_state_str;                         break;
474        case ':':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonField(parser);       break;
475        case ',':   ++parser.ind; parser.state = lib_json_FParser_state_ws; JsonNextElem(parser);    break;
476        default :
477            if (lib_json::JsonFirstNumCharQ(parser.buf[parser.ind])) {
478                parser.state = lib_json_FParser_state_number;
479            } else                                           { parser.state = lib_json_FParser_state_token; }
480            break;
481        }
482    }
483}

485// dispatch to parser routine
486static void JsonParseDispatch(lib_json::FParser &parser) {
487    switch (parser.state) {
488    case lib_json_FParser_state_ws        : JsonSkipWs(parser);                break;
489    case lib_json_FParser_state_value     : JsonParseValue(parser);            break;
490    case lib_json_FParser_state_number    : JsonParseNumber(parser);           break;
491    case lib_json_FParser_state_str       : JsonParseString(parser);           break;
492    case lib_json_FParser_state_str_esc   : JsonParseStringEscape(parser);     break;
493    case lib_json_FParser_state_str_uesc  : JsonParseStringUnicode(parser);    break;
494    case lib_json_FParser_state_str_usurr1: JsonParseStringSurrogate1(parser); break;
495    case lib_json_FParser_state_str_usurr2: JsonParseStringSurrogate2(parser); break;
496    case lib_json_FParser_state_token     : JsonParseToken(parser);            break;
497    case lib_json_FParser_state_sec_line  : JsonParseSecLine(parser);          break;
498    case lib_json_FParser_state_err       : parser.ind = elems_N(parser.buf);  break; // blackhole
499    default: vrfy_(0);
500    }
501}

503inline bool ScalarNodeQ(lib_json::FNode &node) {
504    return node.type != lib_json_FNode_type_field
505        && node.type != lib_json_FNode_type_array
506        && node.type != lib_json_FNode_type_object;
507}

509// Parses JSON text into tree structure according to ECMA-404
510// can be invoked repeadetly with new data portions (can be used directly as read hook)
511//
512// PARSER    parser handle
513// BUF       input buffer to process
514//           non-zero-length - parse new data portion
515//           zero-length     - to finalize and check completeness
516// RETURN    true for OK, false for parse error
517//
518// after finalization, root node pointer is available at: parser.root_node
519// whole tree is automatically deleted upon parser destruction,
520// to detach tree from parser do parser.root_node=NULL
521bool lib_json::JsonParse(lib_json::FParser &parser, strptr buf) {
522    parser.buf = buf;
523    parser.ind = 0;
524    do {
525        // at least one time to finalize
526        JsonParseDispatch(parser);
527    } while (parser.ind<elems_N(parser.buf));
528    // check final state
529    bool ok = elems_N(parser.buf)
530        || (parser.state == lib_json_FParser_state_ws
531            && (!parser.node  || (parser.node == parser.root_node && ScalarNodeQ(*parser.node))));
532    if (!ok) {
533        JsonParseError(parser,"unexpected end of input");
534    }
535    // advance offset
536    parser.offset += elems_N(parser.buf);
537    return lib_json::JsonParseOkQ(parser);
538}

540// AMC cleanup function - automatically delete parsed JSON tree
541void lib_json::root_node_Cleanup(lib_json::FParser& parent) {
542    if (parent.root_node) {
543        node_Delete(*parent.root_node);
544        parent.root_node = NULL;
545    }
546}

548// encode json string
549// "The representation of strings is similar to conventions used in the C
550// family of programming languages.  A string begins and ends with
551// quotation marks.  All Unicode characters may be placed within the
552// quotation marks except for the characters that must be escaped:
553// quotation mark, reverse solidus, and the control characters (U+0000
554// through U+001F)."
555//    -- this says that solidus need not be escaped when printing -- only when parsing!
556void lib_json::JsonSerializeString(algo::strptr str, algo::cstring &lhs) {
557    lhs << '"';
558    frep_(i,elems_N(str)) {
559        switch (str[i]) {
560        case '"' :  lhs << '\\' << str[i]; break;
561        case '\\':  lhs << '\\' << str[i]; break;
562        case '\b':  lhs << "\\b";          break;
563        case '\f':  lhs << "\\f";          break;
564        case '\n':  lhs << "\\n";          break;
565        case '\r':  lhs << "\\r";          break;
566        case '\t':  lhs << "\\t";          break;
567        default  :  if (str[i] & ~0x1f) {
568                lhs << str[i];
569            } else {
570                // escape control chars
571                lhs << "\\u";
572                u64_PrintHex(str[i],lhs,4,false,true);
573            }
574            break;
575        }
576    }
577    lhs << '"';
578}
579
580// Serialize to string
581// Serialize to JSON tree to text
582// NODE      root node to start from
583// LHS       target string
584// PRETTY    whether or not pretty-format
585// INDENT    level of indenting (for pretty-formatting)
586void lib_json::JsonSerialize(lib_json::FNode* node, cstring &lhs, bool pretty, u32 indent) {
587    if (node) {
588        switch (node->type) {
589        case lib_json_FNode_type_null  :    lhs<<"null";      break;
590        case lib_json_FNode_type_true  :    lhs<<"true";      break;
591        case lib_json_FNode_type_false :    lhs<<"false";     break;
592        case lib_json_FNode_type_number:    lhs<<node->value; break;
593
594        case lib_json_FNode_type_string: {
595            JsonSerializeString(node->value,lhs);
596            break;
597        }
598
599        case lib_json_FNode_type_field : {
600            // silently omit malformed fields
601            if (c_child_N(*node)==1) {
602                JsonSerializeString(node->value,lhs);
603                lhs<<':';
604                lib_json::JsonSerialize(c_child_Find(*node,0),lhs,pretty,indent);
605            }
606        } break;
607
608        case lib_json_FNode_type_array:
609        case lib_json_FNode_type_object: {
610            lhs<< (node->type == lib_json_FNode_type_object ? '{' : '[');
611            indent++;
612            algo::ListSep ls(",");
613            ind_beg(lib_json::node_c_child_curs, child, *node) {
614                if (pretty) {
615                    lhs<< '\n';
616                    char_PrintNTimes(' ',lhs,indent*4);
617                }
618                lhs<<ls;
619                lib_json::JsonSerialize(&child,lhs,pretty,indent);
620            }ind_end;
621            indent--;
622            if (pretty) {
623                lhs<<'\n';
624                char_PrintNTimes(' ',lhs,indent*4);
625            }
626            lhs<< (node->type == lib_json_FNode_type_object ? '}' : ']');
627        } break;
628
629        default:
630            vrfy_(0);
631        }
632    }
633}
634
635void lib_json::JsonSerialize(lib_json::FNode* node, cstring &lhs, bool pretty) {
636    lib_json::JsonSerialize(node,lhs,pretty,0);
637}

639void lib_json::JsonSerialize(lib_json::FNode* node, cstring &lhs) {
640    lib_json::JsonSerialize(node,lhs,false,0);
641}

643// Find node in object chain
644// PARENT    node to start from
645// PATH      dot-separated list of field keys
646//           e.g. path abc.def.ghi and parent pointing to {"abc:{"def":{"ghi":value}}}
647//           yields value node
648lib_json::FNode* lib_json::node_Find(lib_json::FNode* parent, strptr path) {
649    algo::StringIter it(path);
650    while (!it.EofQ()) {
651        if (!parent || parent->type!=lib_json_FNode_type_object) {
652            return NULL;
653        }
654        parent = ind_objfld_Find(lib_json::FldKey(parent,GetTokenChar(it,'.')));
655    }
656    return parent;
657}

659// Get node value as string
660// Empty string is returned in case of any error or null value
661// true/false is converted to 0/1
662//
663// PARENT    node to start from
664// PATH      dot-separated list of field keys
665strptr lib_json::strptr_Get(lib_json::FNode* parent, strptr path) {
666    lib_json::FNode* node = lib_json::node_Find(parent,path);
667    strptr ret;
668    if (node) {
669        switch (node->type) {
670        case lib_json_FNode_type_true  :   ret = "1"           ; break;
671        case lib_json_FNode_type_false :   ret = "0"           ; break;
672        case lib_json_FNode_type_string:
673        case lib_json_FNode_type_number:   ret = node->value   ; break;
674        default: break;
675        }
676    }
677    return ret;
678}

680// Get array node, NULL in case of any error
681//
682// PARENT    node to start from
683// PATH      dot-separated list of field keys
684lib_json::FNode* lib_json::node_GetArray(lib_json::FNode* parent, strptr path) {
685    lib_json::FNode* node = node_Find(parent,path);
686    return (!node || node->type != lib_json_FNode_type_array) ? NULL : node;
687}

689// Get node value as u32, EXCEPTION on any error
690// true/false is converted to 0/1
691//
692// PARENT    node to start from
693// PATH      dot-separated list of field keys
694u32 lib_json::u32_Get(lib_json::FNode* parent, strptr path) {
695    lib_json::FNode* node = lib_json::node_Find(parent,path);
696    vrfy(node,"u32_Get: path is not found");
697    u32 ret=0;
698    if (node) {
699        switch (node->type) {
700        case lib_json_FNode_type_true  :   ret = 1;                             break;
701        case lib_json_FNode_type_false :   ret = 0;                             break;
702        case lib_json_FNode_type_number:
703            vrfy(u32_ReadStrptrMaybe(ret,node->value), algo_lib::_db.errtext);
704            break;
705        default:                           vrfy(0,"u32_Get: non-numeric node"); break;
706        }
707    }
708    return ret;
709}

805    lib_json::FldKey ret;
