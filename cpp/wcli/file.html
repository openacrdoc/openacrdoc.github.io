 19    wcli::LoadWikins();

 21    out << "TITLE\tWIKINS\tSTATUS\n";
 22    ind_beg(wcli::_db_pagetitle_curs, pagetitle, wcli::_db) if (pagetitle.p_page){

 24        if (wcli::FWikins *wikins=wcli::ind_wikins_id_Find(pagetitle.wikins)){
 25            wikins_key=wikins->wikins;
 26        } else {
 27            wikins_key=pagetitle.wikins;

 29        out << pagetitle.pagetitle
 30            << "\t" << wikins_key
 31            << "\t" << pagetitle.p_page->status

 33    }ind_end;
 34    prlog(Tabulated(out,"\t"));

142static void DownloadPages(){
143    if (wcli::_db.download_show_only){

157            } else {
158                StringToFile(Trimmed(ftypefile.text_out),name_Get(ftypefile));

164void wcli::wtblact_page_download(wcli::FWtblact &wtblact){
165    vrfy(bool_ReadStrptrMaybe(wcli::_db.download_show_only
166                              ,wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_download,wclidb_Wfld_wfld_show_only))
167         ,"comment show_only value have to be bool");
168    tempstr special_export(wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_download,wclidb_Wfld_wfld_special_export));
169    // Collect pages
170    // Verify that page is not a section
171    vrfy(Replace(wtblact.id,".",".")<2,tempstr()
172         <<Keyval("page",wtblact.id)
173         <<Keyval("comment","can't download a section - has to be the whole page or pages")
174         );
175    Run_PageList(wtblact);
176    if (special_export==""){
177        bool image_list=false;
178        ind_beg(wcli::_db_page_curs,page,wcli::_db){
179            image_list|=wcli::ind_wikins_id_FindX(page.wikins).wikins==wclidb_Wfld_wfld_file;
180        }ind_end;
181        if (image_list){
182            Run_Imagelist();
183        }
184        if (!wcli::_db.download_show_only){
185            Run_PageDetail(true);
186        }
187    }
188    // Form ftypefile
189    wcli::FFtypefile *ftypefile=NULL;
190    if (special_export!=""){
191        ftypefile=&ftypefile_GetOrCreate(wclidb_Ftype_ftype_html,algo::DirFileJoin(wcli::_db.cmdline.pagedir,special_export));
192        if (!wcli::_db.download_show_only){
193            ftypefile->text_out=SpecialExport();
194        }
195    }
196    ind_beg(wcli::_db_page_curs,page,wcli::_db){
197        if (special_export==""){
198            ftypefile=&ftypefile_GetOrCreate(wclidb_Ftype_ftype_page,algo::DirFileJoin(wcli::_db.cmdline.pagedir,Urlencode(page.title)));
199            if (!wcli::_db.download_show_only){
200                ExtractText(page);
201                ftypefile->text_out=page.wikitext;
202            }
203        }
204        title_Alloc(*ftypefile)<<page.title;
205        if (page.image_url!=""){
206            tempstr title(Pathcomp(page.image_url,"/RR"));
207            ftypefile=&ftypefile_GetOrCreate(wclidb_Ftype_ftype_image,algo::DirFileJoin(wcli::_db.cmdline.pagedir,Urlencode(title)));
208            ftypefile->image_url=page.image_url;
209            title_Alloc(*ftypefile)<<title;
210        }
211    }ind_end;
212
213    DownloadPages();
214    Main_ShowFtypelist();
215}

329void wcli::wtblact_page_upload(wcli::FWtblact &){
330    vrfy(bool_ReadStrptrMaybe(wcli::_db.upload_show_only
331                              ,wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_upload,wclidb_Wfld_wfld_show_only))
332         ,"comment show_only value have to be bool");
333    tempstr file_in(wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_upload,wclidb_Wfld_wfld_file));
334    algo_lib::Regx file_regx;
335    Regx_ReadSql(file_regx,wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_upload,wclidb_Wfld_wfld_file), false);
336    algo_lib::Regx type_regx;
337    Regx_ReadSql(type_regx,wcli::GetTblactfld(wclidb_Wtblact_wtblact_page_upload,wclidb_Wfld_wfld_type), false);
338    tempstr file_list;
339    ind_beg(algo::Dir_curs,file,DirFileJoin(wcli::_db.cmdline.pagedir,"*")) {
340        // select file name
341        if (algo_lib::Regx_Match(file_regx,file.filename)
342            && !EndsWithQ(file.filename,"~")
343            ){
344            file_list << file.pathname << eol;
345        }
346    }ind_end;
347    vrfy(file_list!="",tempstr()
348         <<Keyval("pagedir",wcli::_db.cmdline.pagedir)
349         <<Keyval("file",file_in)
350         <<Keyval("comment","no files match the file selector in the directort")
351         );
352    // decide on the file types
353    wcli::GetSiteinfo();
354    wcli::Main_CurlExec();
355    // finally, go work on the files
356
357    TempfileInitX(wcli::_db.tempfile,"wcli");
358    tempstr cmd;
359    cmd<<"cat << 'EOF' | file -f - >" << wcli::_db.tempfile.filename;
360    cmd << eol << file_list;
361    cmd << "EOF";
362    SysCmd(cmd,FailokQ(true),DryrunQ(wcli::_db.cmdline.dry_run),EchoQ(algo_lib::_db.cmdline.verbose));
363
364    ind_beg(algo::FileLine_curs,line,wcli::_db.tempfile.filename) if (line!=""){
365        tempstr path_name(Pathcomp(line,":LL"));
366        tempstr file_type(Trimmed(Pathcomp(line,":LR")));
367        ind_beg(algo::Sep_curs,type,file_type,' ') if (line!=""){
368            MakeLower(type);
369            // deliberate on the file type
370            wcli::FFtype *ftype=wcli::ind_ftype_Find(type);
371            if (!ftype){
372                if (ind_wikiext_Find(type)){
373                    ftype=&wcli::ind_ftype_FindX(wclidb_Ftype_ftype_image);
374                } else {
375                    ftype=&wcli::ind_ftype_FindX(wclidb_Ftype_ftype_page);
376                }
377            }
378            if (algo_lib::Regx_Match(type_regx,ftype->ftype)){
379                wcli::FFtypefile &ftypefile=ftypefile_GetOrCreate(ftype->ftype,path_name);
380                ftypefile.comment=file_type;
381                // title for all but html
382                if (ftype->ftype!=wclidb_Ftype_ftype_html){
383                    title_Alloc(ftypefile)<<wcli::Urldecode(Pathcomp(path_name,"/RR"));
384                }
385            }
386            break;
387        }ind_end;
388    }ind_end;
389    // recover titles for special:upload
390    ind_beg(wcli::_db_ftypefile_curs,ftypefile,wcli::_db) if (ftype_Get(ftypefile)==wclidb_Ftype_ftype_html){
391        cmd=tempstr()<<"grep '<title>' "<<name_Get(ftypefile)<<" | cut -d'>' -f2 | cut -d'<' -f1  >"<<wcli::_db.tempfile.filename;
392        SysCmd(cmd,FailokQ(true),DryrunQ(wcli::_db.cmdline.dry_run),EchoQ(algo_lib::_db.cmdline.verbose));
393        ind_beg(algo::FileLine_curs,line,wcli::_db.tempfile.filename) if (line!=""){
394            title_Alloc(ftypefile)<<line;
395        }ind_end;
396    }ind_end;
397
398    if (!wcli::_db.upload_show_only){
399        ind_beg(wcli::_db_ftype_curs,ftype,wcli::_db){
400            ind_beg(wcli::ftype_c_ftypefile_curs,ftypefile,ftype){
401                // need new token for each upload
402                wcli::GetCsrfToken();
403                wcli::Main_CurlExec();
404                ftype.step(ftypefile);
405            }ind_end;
406        }ind_end;
407        Main_ShowPagetitlelist();
408    } else {
409        Main_ShowFtypelist();
410    }
